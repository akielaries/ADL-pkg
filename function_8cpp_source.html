148 double gpmp::optim::Func::cubic_interpolation(double x,
149  double x0,
150  double x1,
151  double y0,
152  double y1,
153  double y0_prime,
154  double y1_prime) {
155  double h = x1 - x0;
156  double t = (x - x0) / h;
157  double t2 = t * t;
158  double t3 = t2 * t;
159 
160  double a = 2 * t3 - 3 * t2 + 1;
161  double b = t3 - 2 * t2 + t;
162  double c = t3 - t2;
163  double d = -2 * t3 + 3 * t2;
164 
165  return a * y0 + b * (h * y0_prime) + c * (h * y1_prime) + d * y1;

168 double gpmp::optim::Func::golden_section_search_minimize(
169  const std::function<double(double)> &func,
170  double a,
171  double b,
172  double tol,
173  double x1,
174  double x2) {
175  double f1 = func(x1);
176  double f2 = func(x2);
177 
178  while ((b - a) > tol) {
179  if (f1 < f2) {
180  b = x2;
181  x2 = x1;
182  x1 = a + b - x2;
183  f2 = f1;
184  f1 = func(x1);
185  } else {
186  a = x1;
187  x1 = x2;
188  x2 = a + b - x1;
189  f1 = f2;
190  f2 = func(x2);
191  }
192  }
193 
194  return (f1 < f2) ? x1 : x2;
195 }

265 gpmp::optim::Func::fit_linear(const std::vector<double> &x,
266  const std::vector<double> &y) {
267  size_t n = x.size();
268 
269  if (n != y.size() || n < 2) {
270  throw std::invalid_argument(
271  "Invalid input dimensions for linear curve fitting.");
272  }
273 
274  double sum_x = 0.0, sum_y = 0.0, sum_xy = 0.0, sum_x_squared = 0.0;
275 
276  for (size_t i = 0; i < n; ++i) {
277  sum_x += x[i];
278  sum_y += y[i];
279  sum_xy += x[i] * y[i];
280  sum_x_squared += x[i] * x[i];
281  }
282 
283  double a =
284  (n * sum_xy - sum_x * sum_y) / (n * sum_x_squared - sum_x * sum_x);

287  return {a, b};

415 std::vector<double> gpmp::optim::Func::regula_falsi(
416  const std::function<double(const std::vector<double> &)> &func,
417  double lower_bound,
418  double upper_bound,
419  size_t max_iterations) {
420  if (func({lower_bound}) * func({upper_bound}) >= 0.0) {
421  throw std::invalid_argument("Invalid bounds for regula falsi method.");
422  }
423 
424  double a = lower_bound;
425  double b = upper_bound;
426 
427  for (size_t iteration = 0; iteration < max_iterations; ++iteration) {
428  double fa = func({a});
429  double fb = func({b});
430 
431  if (fa * fb >= 0.0) {
432  throw std::runtime_error("Regula falsi method: Function values at "
433  "the bounds have the same sign.");
434  }
435 
436  double c = (a * fb - b * fa) / (fb - fa);
437 
438  if (fabs(func({c}) - 0.0f) < std::numeric_limits<double>::epsilon()) {
439  return {c};
440  } else if (fa * func({c}) < 0.0) {
441  b = c;
442  } else {
443  a = c;
444  }
445  }
446 
447  return {(a + b) / 2.0};
448 }

450 std::vector<double> gpmp::optim::Func::cubic_fit(const std::vector<double> &x,
451  const std::vector<double> &y) {
452  size_t n = x.size();
453 
454  if (n != y.size() || n < 4) {
455  throw std::invalid_argument(
456  "Invalid input dimensions for cubic curve fitting.");
457  }
458 
459  double sum_x = 0.0, sum_y = 0.0, sum_x_squared = 0.0, sum_x_cubed = 0.0,
460  sum_xy = 0.0, sum_x_squared_y = 0.0, sum_squared = 0.0;
461 
462  for (size_t i = 0; i < n; ++i) {
463  double xi_squared = x[i] * x[i];
464  double xi_cubed = xi_squared * x[i];
465 
466  sum_x += x[i];
467  sum_y += y[i];
468  sum_x_squared += xi_squared;
469  sum_x_cubed += xi_cubed;
470  sum_xy += x[i] * y[i];
471  sum_x_squared_y += xi_squared * y[i];
472  }
473 
474  double determinant = n * sum_x_squared * sum_x_squared * sum_x_squared -
475  sum_x_squared * sum_x_squared * sum_x * sum_x_squared -
476  sum_x * sum_x * sum_x_squared * sum_x_squared +
477  sum_x_squared * sum_x * sum_x * sum_x_squared +
478  sum_x * sum_x * sum_x * sum_x_squared -
479  n * sum_x * sum_x * sum_x * sum_x;
480 
481  double a = (n * sum_x_squared * sum_x_squared_y -
482  sum_x_squared * sum_x * sum_xy * sum_x_squared -
483  sum_x * sum_x_squared_y * sum_x_squared +
484  sum_x_squared * sum_x * sum_x * sum_xy +
485  sum_x * sum_x * sum_x_squared * sum_x * sum_y -
486  n * sum_x * sum_x * sum_x * sum_xy) /
487  determinant;
488 
489  double b = (n * sum_x_squared * sum_x_squared * sum_xy -
490  sum_x_squared_y * sum_x_squared * sum_x_squared +
491  sum_x * sum_x * sum_x_squared_y * sum_x_squared -
492  sum_x_squared * sum_x * sum_x * sum_x * sum_xy -
493  sum_x_squared * sum_x_squared * sum_x * sum_y +
494  n * sum_x * sum_x * sum_x * sum_x * sum_xy) /
495  determinant;
496 
497  double c =
498  (n * sum_x_squared * sum_x_squared * sum_x * sum_xy -
499  sum_x_squared * sum_x_squared_y * sum_x_squared * sum_x +
500  sum_x * sum_x * sum_x_squared_y * sum_x_squared * sum_x_squared -
501  sum_x_squared * sum_x * sum_x * sum_x * sum_x_squared -
502  sum_x * sum_x_squared * sum_x_squared * sum_x * sum_y +
503  n * sum_x * sum_x * sum_x * sum_squared * sum_x * sum_y) /
504  determinant;
505 
506  double d =
507  (n * sum_x_squared * sum_x_squared * sum_x * sum_x * sum_xy -
508  sum_x_squared * sum_x_squared_y * sum_x_squared * sum_x_squared *
509  sum_x +
510  sum_x_squared * sum_x * sum_x_squared_y * sum_x_squared *
511  sum_x_squared -
512  sum_x_squared * sum_x * sum_x * sum_x * sum_x_squared -
513  sum_x * sum_x_squared * sum_x_squared * sum_x * sum_x_squared +
514  n * sum_x * sum_x * sum_x * sum_squared * sum_x * sum_x_squared) /
515  determinant;
516 
517  return {a, b, c, d};
518 }

520 std::vector<double> gpmp::optim::Func::nelder_mead(
521  const std::function<double(const std::vector<double> &)> &func,
522  std::vector<double> initial_point,
523  double tolerance,
524  size_t max_iterations) {
525  size_t n = initial_point.size();
526  std::vector<std::vector<double>> simplex(n + 1, initial_point);
527 
528  for (size_t i = 0; i < n; ++i) {
529  simplex[i][i] += 1.0;
530  }
531 
532  std::vector<double> values(n + 1);
533  for (size_t i = 0; i <= n; ++i) {
534  values[i] = func(simplex[i]);
535  }
536 
537  for (size_t iteration = 0; iteration < max_iterations; ++iteration) {
538  size_t highest_index =
539  std::distance(values.begin(),
540  std::max_element(values.begin(), values.end()));
541  size_t lowest_index =
542  std::distance(values.begin(),
543  std::min_element(values.begin(), values.end()));
544 
545  std::vector<double> centroid =
546  calculate_centroid(simplex, highest_index);
547 
548  // Reflect
549  std::vector<double> reflected_point =
550  reflect(simplex[highest_index], centroid, 1.0);
551  double reflected_value = func(reflected_point);
552 
553  if (reflected_value < values[lowest_index]) {
554  // Expand
555  std::vector<double> expanded_point =
556  reflect(simplex[highest_index], centroid, 2.0);
557  double expanded_value = func(expanded_point);
558 
559  if (expanded_value < reflected_value) {
560  simplex[highest_index] = expanded_point;
561  values[highest_index] = expanded_value;
562  } else {
563  simplex[highest_index] = reflected_point;
564  values[highest_index] = reflected_value;
565  }
566  } else if (reflected_value >= values[lowest_index] &&
567  reflected_value < values[highest_index]) {
568  simplex[highest_index] = reflected_point;
569  values[highest_index] = reflected_value;
570  } else {
571  // Contract
572  std::vector<double> contracted_point =
573  reflect(simplex[highest_index], centroid, 0.5);
574  double contracted_value = func(contracted_point);
575 
576  if (contracted_value < values[highest_index]) {
577  simplex[highest_index] = contracted_point;
578  values[highest_index] = contracted_value;
579  } else {
580  // Shrink
581  for (size_t i = 0; i <= n; ++i) {
582  if (i != lowest_index) {
583  for (size_t j = 0; j < n; ++j) {
584  simplex[i][j] = 0.5 * (simplex[i][j] +
585  simplex[lowest_index][j]);
586  }
587  values[i] = func(simplex[i]);
588  }
589  }
590  }
591  }
592 
593  // Check convergence
594  double range = calculate_range(values);
595  if (range < tolerance) {
596  return simplex[lowest_index];
597  }
598  }
599 
600  return simplex[std::distance(
601  values.begin(),
602  std::min_element(values.begin(), values.end()))];
603 }
