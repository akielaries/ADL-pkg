74 gpmp::ml::AutoEncoder::forward(const std::vector<double> &input) {
75  // forward passes
76  std::vector<double> hidden(hidden_size);
77  std::vector<double> output(output_size);
78 
79  // calculate hidden layer values
80  for (int i = 0; i < hidden_size; ++i) {
81  hidden[i] = 0;
82  for (int j = 0; j < input_size; ++j) {
83  hidden[i] += input[j] * weights_input_hidden[j][i];
84  }
85  hidden[i] = sigmoid({hidden[i]})[0];
86  }
87 
88  // calculate output layer values
89  for (int i = 0; i < output_size; ++i) {
90  output[i] = 0;
91  for (int j = 0; j < hidden_size; ++j) {
92  output[i] += hidden[j] * weights_hidden_output[j][i];
93  }
94  output[i] = sigmoid({output[i]})[0];
95  }

100 void gpmp::ml::AutoEncoder::train(
101  const std::vector<std::vector<double>> &training_data,
102  int epochs) {
103  for (int epoch = 0; epoch < epochs; ++epoch) {
104  for (const auto &input : training_data) {
105  // forward pass
106  std::vector<double> hidden(hidden_size);
107  std::vector<double> output(output_size);
108 
109  // calculate hidden layer values
110  for (int i = 0; i < hidden_size; ++i) {
111  hidden[i] = 0;
112  for (int j = 0; j < input_size; ++j) {
113  hidden[i] += input[j] * weights_input_hidden[j][i];
114  }
115  hidden[i] = sigmoid({hidden[i]})[0];
116  }
117 
118  // calculate output layer values
119  for (int i = 0; i < output_size; ++i) {
120  output[i] = 0;
121  for (int j = 0; j < hidden_size; ++j) {
122  output[i] += hidden[j] * weights_hidden_output[j][i];
123  }
124  output[i] = sigmoid({output[i]})[0];
125  }
126 
127  // backward pass (gradient descent)
128  for (int i = 0; i < output_size; ++i) {
129  for (int j = 0; j < hidden_size; ++j) {
130  weights_hidden_output[j][i] -=
131  learning_rate * (output[i] - input[i]) * hidden[j];
132  }
133  }
134 
135  for (int i = 0; i < hidden_size; ++i) {
136  for (int j = 0; j < input_size; ++j) {
137  double error = 0;
138  for (int k = 0; k < output_size; ++k) {
139  error += (output[k] - input[k]) *
140  weights_hidden_output[i][k];
141  }
142  weights_input_hidden[j][i] -= learning_rate * error *
143  input[j] * (1 - hidden[i]) *
144  hidden[i];
145  }
146  }
147  }
148  }
149 }
