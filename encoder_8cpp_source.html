160  std::vector<double> output(output_size);
161

166  hidden[i] += input[j] * weights_input_hidden[j][i];

170 
171  // calculate output layer values
172  for (int i = 0; i < output_size; ++i) {
173  output[i] = 0;
174  for (int j = 0; j < hidden_size; ++j) {
175  output[i] += hidden[j] * weights_hidden_output[j][i];
176  }
177  output[i] = sigmoid({output[i]})[0];

179 
180  // backward pass (gradient descent)
181  for (int i = 0; i < output_size; ++i) {
182  for (int j = 0; j < hidden_size; ++j) {
183  weights_hidden_output[j][i] -=
184  learning_rate * (output[i] - input[i]) * hidden[j];
185  }
186  }
187

189  for (int j = 0; j < input_size; ++j) {
190  double error = 0;
191  for (int k = 0; k < output_size; ++k) {
192  error += (output[k] - input[k]) *
193  weights_hidden_output[i][k];
194  }
195  weights_input_hidden[j][i] -= learning_rate * error *
196  input[j] * (1 - hidden[i]) *

232 void gpmp::ml::SparseAutoEncoder::train(
233  const std::vector<std::vector<double>> &training_data,
234  int epochs) {
235 
236  const double SPARSITY_TARGET_DECAY = 0.1;
237 
238  for (int epoch = 0; epoch < epochs; ++epoch) {
239  for (const auto &current_input : training_data) {
240  // forward pass
241  std::vector<double> hidden = forward(current_input);
242 
243  // backward pass (gradient descent)
244  for (int i = 0; i < output_size; ++i) {
245  for (int j = 0; j < hidden_size; ++j) {
246  weights_hidden_output[j][i] -=
247  learning_rate * (hidden[i] - current_input[i]) *
248  hidden[j];
249  }
250  }
251 
252  for (int i = 0; i < hidden_size; ++i) {
253  for (int j = 0; j < input_size; ++j) {
254  double error = 0;
255  for (int k = 0; k < output_size; ++k) {
256  error += (hidden[k] - current_input[k]) *
257  weights_hidden_output[i][k];
258  }
259 
260  double sparsity_term =
261  sparsity_weight * (sparsity_target - hidden[i]);
262 
263  weights_input_hidden[j][i] -=
264  learning_rate * (error + sparsity_term) *
265  current_input[j] * (1 - hidden[i]) * hidden[i];
266  }
267  }
268 
269  for (int i = 0; i < hidden_size; ++i) {
270  double average_activation = 0.0;
271  for (const auto &input : training_data) {
272  std::vector<double> current_hidden = forward(input);
273  average_activation += current_hidden[i];
274  }
275  average_activation /= training_data.size();
276  sparsity_target =
277  (1.0 - SPARSITY_TARGET_DECAY) * sparsity_target +
278  SPARSITY_TARGET_DECAY * average_activation;
279  }
280  }
281  }
282 }

433 
434  for (int epoch = 0; epoch < epochs; ++epoch) {

520 void gpmp::ml::VariationalAutoEncoder::gradient_descent(
521  const std::vector<double> &input,
522  const std::vector<double> &output,
523  const std::vector<double> &hidden_sampled) {
524  for (int i = 0; i < output_size; ++i) {
525  for (int j = 0; j < hidden_size; ++j) {
526  weights_hidden_output[j][i] -=
527  learning_rate * (output[i] - input[i]) * hidden_sampled[j];
528  }
529  }
530 
531  for (int i = 0; i < hidden_size; ++i) {
532  for (int j = 0; j < input_size; ++j) {
533  double error = 0;
534  for (int k = 0; k < output_size; ++k) {
535  error += (output[k] - input[k]) * weights_hidden_output[i][k];
536  }
537  double hidden_gradient =
538  hidden_sampled[i] * (1 - hidden_sampled[i]);
539  // derivative of softplus
540  double log_variance_gradient =
541  1.0 / (1.0 + exp(-hidden_log_variance[i]));
542  weights_input_hidden[j][i] -=
543  learning_rate *
544  (error * hidden_gradient +
545  (hidden_sampled[i] - hidden_mean[i]) * hidden_gradient +
546  (hidden_log_variance[i] - log_variance_gradient) *
547  hidden_gradient) *
548  input[j];
549  }
550  }
551 }

553 void gpmp::ml::VariationalAutoEncoder::train(
554  const std::vector<std::vector<double>> &training_data,
555  int epochs) {
556  std::default_random_engine generator;
557  std::normal_distribution<double> normal_distribution(0.0, 1.0);
558  std::vector<double> hidden_sampled;
559 
560  for (int epoch = 0; epoch < epochs; ++epoch) {
561  for (const auto &input : training_data) {
562  // forward pass (encoder)
563  hidden_mean = encoder(input);
564  hidden_log_variance = encoder(input);
565 
566  for (int i = 0; i < hidden_size; ++i) {
567  hidden_sampled.push_back(
568  reparameterize(hidden_mean[i], hidden_log_variance[i]));
569  }
570 
571  // forward pass (decoder)
572  std::vector<double> output = decoder(hidden_sampled);
573 
574  // backward pass (gradient descent)
575  gradient_descent(input, output, hidden_sampled);
576  }
577  }

634 std::vector<double> gpmp::ml::RecurrentAutoEncoder::recurr_fwd(
635  const std::vector<double> &input,
636  const std::vector<double> &previous_hidden) {
637  std::vector<double> recurrent_input(hidden_size, 0.0);
638 
639  // sum the weighted contributions from the current input and the previous
