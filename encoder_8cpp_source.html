112 std::vector<double>
113 gpmp::ml::AutoEncoder::forward(const std::vector<double> &input) {
114  // forward passes
115  std::vector<double> hidden(hidden_size);
116  std::vector<double> output(output_size);
117 
118  // calculate hidden layer values
119  for (int i = 0; i < hidden_size; ++i) {
120  hidden[i] = 0;
121  for (int j = 0; j < input_size; ++j) {
122  hidden[i] += input[j] * weights_input_hidden[j][i];
123  }
124  hidden[i] = sigmoid({hidden[i]})[0];
125  }
126 
127  // calculate output layer values
128  for (int i = 0; i < output_size; ++i) {
129  output[i] = 0;
130  for (int j = 0; j < hidden_size; ++j) {
131  output[i] += hidden[j] * weights_hidden_output[j][i];
132  }
133  output[i] = sigmoid({output[i]})[0];
134  }
135 
136  return output;
137 }

139 void gpmp::ml::AutoEncoder::train(
140  const std::vector<std::vector<double>> &training_data,
141  int epochs) {
142  for (int epoch = 0; epoch < epochs; ++epoch) {
143  for (const auto &input : training_data) {
144  // forward pass
145  std::vector<double> hidden(hidden_size);
146  std::vector<double> output(output_size);
147 
148  // calculate hidden layer values
149  for (int i = 0; i < hidden_size; ++i) {
150  hidden[i] = 0;
151  for (int j = 0; j < input_size; ++j) {
152  hidden[i] += input[j] * weights_input_hidden[j][i];
153  }
154  hidden[i] = sigmoid({hidden[i]})[0];
155  }
156 
157  // calculate output layer values
158  for (int i = 0; i < output_size; ++i) {
159  output[i] = 0;
160  for (int j = 0; j < hidden_size; ++j) {
161  output[i] += hidden[j] * weights_hidden_output[j][i];
162  }
163  output[i] = sigmoid({output[i]})[0];
164  }
165 
166  // backward pass (gradient descent)
167  for (int i = 0; i < output_size; ++i) {
168  for (int j = 0; j < hidden_size; ++j) {
169  weights_hidden_output[j][i] -=
170  learning_rate * (output[i] - input[i]) * hidden[j];
171  }
172  }
173 
174  for (int i = 0; i < hidden_size; ++i) {
175  for (int j = 0; j < input_size; ++j) {
176  double error = 0;
177  for (int k = 0; k < output_size; ++k) {
178  error += (output[k] - input[k]) *
179  weights_hidden_output[i][k];
180  }
181  weights_input_hidden[j][i] -= learning_rate * error *
182  input[j] * (1 - hidden[i]) *
183  hidden[i];
184  }
185  }
186  }
187  }
188 }

218 void gpmp::ml::SparseAutoEncoder::train(
219  const std::vector<std::vector<double>> &training_data,
220  int epochs) {
221 
222  const double SPARSITY_TARGET_DECAY = 0.1;
223 
224  for (int epoch = 0; epoch < epochs; ++epoch) {
225  for (const auto &current_input : training_data) {
226  // forward pass
227  std::vector<double> hidden = forward(current_input);
228 
229  // backward pass (gradient descent)
230  for (int i = 0; i < output_size; ++i) {
231  for (int j = 0; j < hidden_size; ++j) {
232  weights_hidden_output[j][i] -=
233  learning_rate * (hidden[i] - current_input[i]) *
234  hidden[j];
235  }
236  }
237 
238  for (int i = 0; i < hidden_size; ++i) {
239  for (int j = 0; j < input_size; ++j) {
240  double error = 0;
241  for (int k = 0; k < output_size; ++k) {
242  error += (hidden[k] - current_input[k]) *
243  weights_hidden_output[i][k];
244  }
245 
246  double sparsity_term =
247  sparsity_weight * (sparsity_target - hidden[i]);
248 
249  weights_input_hidden[j][i] -=
250  learning_rate * (error + sparsity_term) *
251  current_input[j] * (1 - hidden[i]) * hidden[i];
252  }
253  }
254 
255  for (int i = 0; i < hidden_size; ++i) {
256  double average_activation = 0.0;
257  for (const auto &input : training_data) {
258  std::vector<double> current_hidden = forward(input);
259  average_activation += current_hidden[i];
260  }
261  average_activation /= training_data.size();
262  sparsity_target =
263  (1.0 - SPARSITY_TARGET_DECAY) * sparsity_target +
264  SPARSITY_TARGET_DECAY * average_activation;
265  }
266  }
267  }
268 }
