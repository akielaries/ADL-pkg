48 void gpmp::ml::LogReg::train(const std::vector<std::vector<double>> &X_train,
49  const std::vector<int> &y_train) {
50  // Initialize weights to zeros
51  weights.assign(X_train[0].size() + 1, 0.0);
52 
53  for (int iter = 0; iter < num_iterations; ++iter) {
54  std::vector<double> gradient(X_train[0].size() + 1, 0.0);
55 
56  for (size_t i = 0; i < X_train.size(); ++i) {
57  // Add bias term to input
58  std::vector<double> input = {1.0};
59  input.insert(input.end(), X_train[i].begin(), X_train[i].end());
60 
61  // Compute the predicted value
62  double predicted = sigmoid(std::inner_product(input.begin(),
63  input.end(),
64  weights.begin(),
65  0.0));
66 
67  // Compute gradient for each weight
68  for (size_t j = 0; j < gradient.size(); ++j) {
69  gradient[j] += (predicted - y_train[i]) * input[j];
70  }
71  }
72 
73  // Update weights using gradient descent
74  for (size_t j = 0; j < weights.size(); ++j) {
75  weights[j] -= learning_rate *
76  (gradient[j] / X_train.size() + lambda * weights[j]);
77  }
78  }
79 }

117 void gpmp::ml::LogReg::feature_scaling(std::vector<std::vector<double>> &X) {
118  if (X.empty()) {
119  throw std::invalid_argument("Input feature matrix is empty.");
120  }
121 
122  size_t num_features = X[0].size();
123  for (size_t j = 0; j < num_features; ++j) {
124  double min_val = X[0][j], max_val = X[0][j];
125  for (size_t i = 1; i < X.size(); ++i) {
126  if (X[i][j] < min_val) {
127  min_val = X[i][j];
128  }
129  if (X[i][j] > max_val) {
130  max_val = X[i][j];
131  }
132  }
133 
134  if (min_val == max_val) {
135  continue; // Skip if all values are the same
136  }
137 
138  double range = max_val - min_val;
139  for (size_t i = 0; i < X.size(); ++i) {
140  X[i][j] = (X[i][j] - min_val) / range;
141  }
142  }
143 }

145 double
146 gpmp::ml::LogReg::cost_function(const std::vector<std::vector<double>> &X,
147  const std::vector<int> &y) {
148  double cost = 0.0;
149  for (size_t i = 0; i < X.size(); ++i) {
150  // Add bias term to input
151  std::vector<double> input = {1.0};
152  input.insert(input.end(), X[i].begin(), X[i].end());
153 
154  double predicted = sigmoid(std::inner_product(input.begin(),
155  input.end(),
156  weights.begin(),
157  0.0));
158  cost += -y[i] * log(predicted) - (1 - y[i]) * log(1 - predicted);
159  }
160  cost /= X.size();
161  return cost;
162 }
