160  explicit SecondaryMLP(std::vector<size_t> layer_units,
161  long double lr = .001) :
162  layer_units(layer_units),
163  wt_mtx(),
164  bias_vectors(),
165  lr(lr) {
166 
167  // traverse the elements
168  for (size_t i = 0; i < layer_units.size() - 1; ++i) {
169  // size of inputs
170  size_t inputs{layer_units[i]};
171  // size of outputs
172  size_t outputs{layer_units[i + 1]};
173 
174  // set to random Guassian Noise related values
175  // weights
176  auto gauss_wt = mtpk::mtx<T>::randn(outputs, inputs);
177  wt_mtx.push_back(gauss_wt);
178  // biases
179  auto bias_wt = mtpk::mtx<T>::randn(outputs, 1);
180  bias_vectors.push_back(bias_wt);
181  // activation function
182  activations.resize(layer_units.size());
183  }
184  }
190  auto prop_forwards(Matrix<T> x) {
191  assert(get<0>(x.shape) == layer_units[0] &&
192  get<1>(x.shape));
193  // input = to previously declared acitvations method
194  activations[0] = x;
195  Matrix prev(x);
196 
197  // traverse layer units
198  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
199  Matrix y = wt_mtx[i].mult(prev);
200  y = y + bias_vectors[i];
201  y = y.apply_function(sigmoid_activ);
202  activations[i + 1] = y;
203  prev = y;
204  }
205  return prev;
206  }
215  void prop_backwards(Matrix<T> target) {
216  assert(get<0>(target.shape) == layer_units.back());
217  // calculate the error, target - ouput
218  auto y = target;
219  auto y_hat = activations.back();
220  auto err = target - y_hat;
221  // back propagate the error calculated from output to input
222  // and step the weights
223  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
224  // calculate errors for previous layer
225  auto wt = wt_mtx[i].T();
226  auto prior_errs = wt.mult(err);
227 
228  auto outputs_d = activations[i+1].apply_function(sigmoid_deriv);
229  auto gradients = err.mult_elem(outputs_d);
230  gradients = gradients.scalar_mult(lr);
231  auto trans_a = activations[i].T();
232  auto gradients_wt = gradients.mult(trans_a);
233 
234  // adjust the networks weights based on propagation technique
235  bias_vectors[i] = bias_vectors[i].add(gradients);
236  wt_mtx[i] = wt_mtx[i].add(gradients_wt);
237  err = prior_errs;
238  }
