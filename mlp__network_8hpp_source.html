161  explicit SecondaryMLP(std::vector<size_t> layer_units,
162  long double lr = .001) :
163  layer_units(layer_units),
164  wt_mtx(),
165  bias_vectors(),
166  lr(lr) {
167 
168  // traverse the elements
169  for (size_t i = 0; i < layer_units.size() - 1; ++i) {
170  // size of inputs
171  size_t inputs{layer_units[i]};
172  // size of outputs
173  size_t outputs{layer_units[i + 1]};
174 
175  // set to random Guassian Noise related values
176  // weights
177  auto gauss_wt = mtpk::mtx<T>::randn(outputs, inputs);
178  wt_mtx.push_back(gauss_wt);
179  // biases
180  auto bias_wt = mtpk::mtx<T>::randn(outputs, 1);
181  bias_vectors.push_back(bias_wt);
182  // activation function
183  activations.resize(layer_units.size());
184  }
185  }
191  auto prop_forwards(Matrix<T> x) {
192  assert(get<0>(x.shape) == layer_units[0] &&
193  get<1>(x.shape));
194  // input = to previously declared acitvations method
195  activations[0] = x;
196  Matrix prev(x);
197 
198  // traverse layer units
199  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
200  Matrix y = wt_mtx[i].mult(prev);
201  y = y + bias_vectors[i];
202  y = y.apply_function(sigmoid_activ);
203  activations[i + 1] = y;
204  prev = y;
205  }
206  return prev;
207  }
216  void prop_backwards(Matrix<T> target) {
217  assert(get<0>(target.shape) == layer_units.back());
218  // calculate the error, target - ouput
219  auto y = target;
220  auto y_hat = activations.back();
221  auto err = target - y_hat;
222  // back propagate the error calculated from output to input
223  // and step the weights
224  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
225  // calculate errors for previous layer
226  auto wt = wt_mtx[i].T();
227  auto prior_errs = wt.mult(err);
228 
229  auto outputs_d = activations[i+1].apply_function(sigmoid_deriv);
230  auto gradients = err.mult_elem(outputs_d);
231  gradients = gradients.scalar_mult(lr);
232  auto trans_a = activations[i].T();
233  auto gradients_wt = gradients.mult(trans_a);
234 
235  // adjust the networks weights based on propagation technique
236  bias_vectors[i] = bias_vectors[i].add(gradients);
237  wt_mtx[i] = wt_mtx[i].add(gradients_wt);
238  err = prior_errs;
239  }
