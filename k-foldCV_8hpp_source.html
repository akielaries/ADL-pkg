81 template <class T>
82 KF_CV<T>::KF_CV(const vector<T> &instance_list, int64_t K, uint64_t seed) {
83  this->instance_list = instance_list;
84  shuffle(this->instance_list.begin(), this->instance_list.end(),
85  default_random_engine(seed));

98 template <class T>
99 vector<T> KF_CV<T>::get_train_fold(int64_t k) {
100  vector<T> train_fold;
101  for (int64_t i = 0; i < (k * N) / this->K; i++) {
102  train_fold.push_back(instance_list.at(i));
103  }
104  for (int64_t i = ((k + 1) * N) / this->K; i < N; i++) {
105  train_fold.push_back(instance_list.at(i));
106  }
107  return train_fold;
108 }

117 template <class T>

121  test_fold.push_back(instance_list.at(i));
122  }
123  return test_fold;
