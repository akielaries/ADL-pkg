546  swig_module_info *end,

568  l = i + 1;
569  }
570  } else {

573  } while (l <= r);
574  }

576  } while (iter != end);

2812 /* -----------------------------------------------------------------------------
2813  * Type initialization:
2814  * This problem is tough by the requirement that no dynamic
2815  * memory is used. Also, since swig_type_info structures store pointers to
2816  * swig_cast_info structures and swig_cast_info structures store pointers back
2817  * to swig_type_info structures, we need some lookup code at initialization.
2818  * The idea is that swig generates all the structures that are needed.
2819  * The runtime then collects these partially filled structures.
2820  * The SWIG_InitializeModule function takes these initial arrays out of
2821  * swig_module, and does all the lookup, filling in the swig_module.types
2822  * array with the correct data and linking the correct swig_cast_info
2823  * structures together.
2824  *
2825  * The generated swig_type_info structures are assigned statically to an initial
2826  * array. We just loop through that array, and handle each type individually.
2827  * First we lookup if this type has been already loaded, and if so, use the
2828  * loaded structure instead of the generated one. Then we have to fill in the
2829  * cast linked list. The cast data is initially stored in something like a
2830  * two-dimensional array. Each row corresponds to a type (there are the same
2831  * number of rows as there are in the swig_type_initial array). Each entry in
2832  * a column is one of the swig_cast_info structures for that type.
2833  * The cast_initial array is actually an array of arrays, because each row has
2834  * a variable number of columns. So to actually build the cast linked list,
2835  * we find the array of casts associated with the type, and loop through it
2836  * adding the casts to the list. The one last trick we need to do is making
2837  * sure the type pointer in the swig_cast_info struct is correct.
2838  *
2839  * First off, we lookup the cast->type name to see if it is already loaded.
2840  * There are three cases to handle:
2841  * 1) If the cast->type has already been loaded AND the type we are adding
2842  * casting info to has not been loaded (it is in this module), THEN we
2843  * replace the cast->type pointer with the type pointer that has already
2844  * been loaded.
2845  * 2) If BOTH types (the one we are adding casting info to, and the
2846  * cast->type) are loaded, THEN the cast info has already been loaded by
2847  * the previous module so we just ignore it.
2848  * 3) Finally, if cast->type has not already been loaded, then we add that
2849  * swig_cast_info to the linked list (because the cast->type) pointer will
2850  * be correct.
2851  * ----------------------------------------------------------------------------- */
