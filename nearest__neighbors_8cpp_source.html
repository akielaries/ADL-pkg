158 bool KNN(vector<Company> &data, Company &test, int k,
159  double (*distanceFunction)(Company &, Company &)) {
160  // filling the distances between all points and test
161  fillDistances(data, test, distanceFunction);
162 
163  // sorting so that we can get the k nearest
164  sort(data.begin(), data.end(), comparison);
165 
166  int64_t countSuccesful = 0;
167  int64_t countUnsuccesful = 0;
168  for (int64_t i = 0; i < k; ++i) {
169  if (data[i].success) {
170  ++countSuccesful;
171  } else {
172  ++countUnsuccesful;
173  }
174  }
175  /*
176  * Based on the count of succesful and unsuccessful examples
177  * within those k, we are deciding whether the test example is
178  * successful or not.
179  */
180  if (countSuccesful == countUnsuccesful) {
181  test.success = data[0].success;
182  } else {
183  test.success = countSuccesful > countUnsuccesful ? true : false;
184  }
185  return test.success;
186 }
