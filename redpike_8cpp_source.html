51 void gpmp::RedPike::encrypt(gpmp::RedPike::word *x,
52  const gpmp::RedPike::word *k) {
53  gpmp::RedPike::word rk0 = k[0];
54  gpmp::RedPike::word rk1 = k[1];
55 
56  for (int i = 0; i < gpmp::RedPike::ROUNDS; i++) {
57  rk0 += gpmp::RedPike::CONST;
58  rk1 -= gpmp::RedPike::CONST;
59 
60  x[0] ^= rk0;
61  x[0] += x[1];
62  x[0] = gpmp::RedPike::ROTL(x[0], x[1]);
63 
64  x[1] = gpmp::RedPike::ROTR(x[1], x[0]);
65  x[1] -= x[0];
66  x[1] ^= rk1;
67  }
68 
69  rk0 = x[0];
70  x[0] = x[1];
71  x[1] = rk0;
72 }
