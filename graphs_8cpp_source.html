90 void gpmp::Graph::dijkstra(int start) {
91  std::priority_queue<std::pair<int, int>,
92  std::vector<std::pair<int, int>>,
93  std::greater<std::pair<int, int>>>
94  pq;
95  std::vector<int> dist(vertices, std::numeric_limits<int>::max());
96 
97  dist[start] = 0;
98  pq.push({0, start});
99 
100  while (!pq.empty()) {
101  int u = pq.top().second;
102  pq.pop();
103 
104  for (const auto &neighbor : adj_list[u]) {
105  int v = neighbor.first;
106  int weight = neighbor.second;
107 
108  if (dist[u] + weight < dist[v]) {
109  dist[v] = dist[u] + weight;
110  pq.push({dist[v], v});
111  }
112  }
113  }
114 
115  std::cout << "Dijkstra's Shortest Paths from vertex " << start << ":\n";
116  for (int i = 0; i < vertices; ++i)

143 std::vector<std::pair<int, std::pair<int, int>>> gpmp::Graph::kruskal() {
144  std::vector<std::pair<int, std::pair<int, int>>> edges;
145  for (int u = 0; u < vertices; ++u) {
146  for (const auto &neighbor : adj_list[u]) {
147  int v = neighbor.first;
148  int weight = neighbor.second;
149  edges.emplace_back(weight, std::make_pair(u, v));
150  }
151  }
152 
153  std::sort(edges.begin(), edges.end());
154 
155  std::vector<std::pair<int, std::pair<int, int>>> min_span_tree;
156  std::vector<int> parent(vertices, -1);
157 
158  for (const auto &edge : edges) {
159  int u = edge.second.first;
160  int v = edge.second.second;
161 
162  int setU = find(parent, u);
163  int setV = find(parent, v);
164 
165  if (setU != setV) {
166  min_span_tree.push_back(edge);
167  union_sets(parent, setU, setV);
168  }
169  }
170 
171  return min_span_tree;
172 }

174 void gpmp::Graph::floyd_warshall() {
175  std::vector<std::vector<int>> dist(
176  vertices,
177  std::vector<int>(vertices, std::numeric_limits<int>::max()));
178 
179  for (int i = 0; i < vertices; ++i) {
180  dist[i][i] = 0;
181  for (const auto &neighbor : adj_list[i]) {
182  int v = neighbor.first;
183  int weight = neighbor.second;
184  dist[i][v] = weight;
185  }
186  }
187 
188  for (int k = 0; k < vertices; ++k) {
189  for (int i = 0; i < vertices; ++i) {
190  for (int j = 0; j < vertices; ++j) {
191  if (dist[i][k] != std::numeric_limits<int>::max() &&
192  dist[k][j] != std::numeric_limits<int>::max() &&
193  dist[i][k] + dist[k][j] < dist[i][j]) {
194  dist[i][j] = dist[i][k] + dist[k][j];
195  }
196  }
197  }
198  }
199 
200  std::cout << "Floyd-Warshall Shortest Paths:\n";
201  for (int i = 0; i < vertices; ++i) {
202  for (int j = 0; j < vertices; ++j) {
203  std::cout << "From " << i << " to " << j << ": ";
204  if (dist[i][j] == std::numeric_limits<int>::max())
205  std::cout << "INF";
206  else
207  std::cout << dist[i][j];
208  std::cout << "\n";
209  }
210  }
211 }
