122  std::cout << std::endl;
123 
124  // Print data rows
125  for (const auto &row : data_) {
126  for (const auto &cell : row) {
127  // Check the type of cell and print accordingly
128  if (std::holds_alternative<int64_t>(cell)) {
129  std::cout << std::get<int64_t>(cell);
130  } else if (std::holds_alternative<long double>(cell)) {
131  std::cout << std::get<long double>(cell);
132  } else if (std::holds_alternative<std::string>(cell)) {
133  std::cout << std::get<std::string>(cell);
134  }
135
