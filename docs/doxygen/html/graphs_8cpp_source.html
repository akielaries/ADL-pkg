92void gpmp::Graph::dijkstra(int start) {
93 std::priority_queue<std::pair<int, int>,
94 std::vector<std::pair<int, int>>,
95 std::greater<std::pair<int, int>>>
96 pq;
97 std::vector<int> dist(vertices, std::numeric_limits<int>::max());
98
99 dist[start] = 0;
100 pq.push({0, start});
101
102 while (!pq.empty()) {
103 int u = pq.top().second;
104 pq.pop();
105
106 for (const auto &neighbor : adj_list[u]) {
107 int v = neighbor.first;
108 int weight = neighbor.second;
109
110 if (dist[u] + weight < dist[v]) {
111 dist[v] = dist[u] + weight;
112 pq.push({dist[v], v});
113 }
114 }
115 }
116
117 std::cout << "Dijkstra's Shortest Paths from vertex " << start << ":\n";
118 for (int i = 0; i < vertices; ++i)
119 std::cout << "Vertex " << i << ": " << dist[i] << "\n";
120}

122void gpmp::Graph::bellman_ford(int start) {
123 std::vector<int> dist(vertices, std::numeric_limits<int>::max());
124 dist[start] = 0;
125
126 for (int i = 0; i < vertices - 1; ++i) {
127 for (int u = 0; u < vertices; ++u) {
128 for (const auto &neighbor : adj_list[u]) {
129 int v = neighbor.first;
130 int weight = neighbor.second;
131
132 if (dist[u] != std::numeric_limits<int>::max() &&
133 dist[u] + weight < dist[v]) {
134 dist[v] = dist[u] + weight;
135 }
136 }
137 }
138 }
139
140 std::cout << "Bellman-Ford Shortest Paths from vertex " << start << ":\n";
141 for (int i = 0; i < vertices; ++i)
142 std::cout << "Vertex " << i << ": " << dist[i] << "\n";
143}

145std::vector<std::pair<int, std::pair<int, int>>> gpmp::Graph::kruskal() {
146 std::vector<std::pair<int, std::pair<int, int>>> edges;
147 for (int u = 0; u < vertices; ++u) {
148 for (const auto &neighbor : adj_list[u]) {
149 int v = neighbor.first;
150 int weight = neighbor.second;
151 edges.emplace_back(weight, std::make_pair(u, v));
152 }
153 }
154
155 std::sort(edges.begin(), edges.end());
156
157 std::vector<std::pair<int, std::pair<int, int>>> min_span_tree;
158 std::vector<int> parent(vertices, -1);
159
160 for (const auto &edge : edges) {
161 int u = edge.second.first;
162 int v = edge.second.second;
163
164 int setU = find(parent, u);
165 int setV = find(parent, v);
166
167 if (setU != setV) {
168 min_span_tree.push_back(edge);
169 union_sets(parent, setU, setV);
170 }
171 }
172
173 return min_span_tree;
174}

176void gpmp::Graph::floyd_warshall() {
177 std::vector<std::vector<int>> dist(
178 vertices,
179 std::vector<int>(vertices, std::numeric_limits<int>::max()));
180
181 for (int i = 0; i < vertices; ++i) {
182 dist[i][i] = 0;
183 for (const auto &neighbor : adj_list[i]) {
184 int v = neighbor.first;
185 int weight = neighbor.second;
186 dist[i][v] = weight;
187 }
188 }
189
190 for (int k = 0; k < vertices; ++k) {
191 for (int i = 0; i < vertices; ++i) {
192 for (int j = 0; j < vertices; ++j) {
193 if (dist[i][k] != std::numeric_limits<int>::max() &&
194 dist[k][j] != std::numeric_limits<int>::max() &&
195 dist[i][k] + dist[k][j] < dist[i][j]) {
196 dist[i][j] = dist[i][k] + dist[k][j];
197 }
198 }
199 }
200 }
201
202 std::cout << "Floyd-Warshall Shortest Paths:\n";
203 for (int i = 0; i < vertices; ++i) {
204 for (int j = 0; j < vertices; ++j) {
205 std::cout << "From " << i << " to " << j << ": ";
206 if (dist[i][j] == std::numeric_limits<int>::max())
207 std::cout << "INF";
208 else
209 std::cout << dist[i][j];
210 std::cout << "\n";
211 }
212 }
213}

275bool gpmp::Graph::is_bipartite() {
276 std::vector<int> color(vertices, -1);
277 std::queue<int> bfs_queue;
278
279 for (int start = 0; start < vertices; ++start) {
280 if (color[start] == -1) {
281 color[start] = 0;
282 bfs_queue.push(start);
283
284 while (!bfs_queue.empty()) {
285 int v = bfs_queue.front();
286 bfs_queue.pop();
287
288 for (const auto &neighbor : adj_list[v]) {
289 int u = neighbor.first;
290 if (color[u] == -1) {
291 color[u] = 1 - color[v];
292 bfs_queue.push(u);
293 } else if (color[u] == color[v]) {
294 return false; // Not bipartite
295 }
296 }
297 }
298 }
299 }
300
301 return true; // Bipartite
302}

304std::vector<double> gpmp::Graph::betweenness_centrality() {
305 std::vector<double> centrality(vertices, 0.0);
306
307 for (int s = 0; s < vertices; ++s) {
308 std::vector<int> predecessors(vertices, -1);
309 std::vector<int> distance(vertices, -1);
310 std::vector<int> num_shortest_paths(vertices, 0);
311 std::stack<int> stack;
312 std::queue<int> bfs_queue;
313
314 distance[s] = 0;
315 num_shortest_paths[s] = 1;
316 bfs_queue.push(s);
317
318 while (!bfs_queue.empty()) {
319 int v = bfs_queue.front();
320 bfs_queue.pop();
321 stack.push(v);
322
323 for (const auto &neighbor : adj_list[v]) {
324 int w = neighbor.first;
325 if (distance[w] == -1) {
326 distance[w] = distance[v] + 1;
327 bfs_queue.push(w);
328 }
329
330 if (distance[w] == distance[v] + 1) {
331 num_shortest_paths[w] += num_shortest_paths[v];
332 predecessors[w] = v;
333 }
334 }
335 }
336
337 std::vector<double> dependency(vertices, 0.0);
338
339 while (!stack.empty()) {
340 int w = stack.top();
341 stack.pop();
342
343 for (const auto &predecessor : adj_list[w]) {
344 int v = predecessor.first;
345 dependency[v] += (static_cast<double>(num_shortest_paths[v]) /
346 num_shortest_paths[w]) *
347 (1 + dependency[w]);
348 }
349
350 if (w != s) {
351 centrality[w] += dependency[w];
352 }
353 }
354 }
355
356 return centrality;
357}

401bool gpmp::Graph::hamiltonian_circuit_util(int v,
402 std::vector<bool> &visited,
403 std::vector<int> &path,
404 int count) {
405 visited[v] = true;
406 path.push_back(v);
407
408 if (count == vertices) {
409 // all vertices are visited; check if there is an edge from the last
410 // added vertex to the starting vertex
411 for (const auto &neighbor : adj_list[v]) {
412 if (neighbor.first == path[0]) {
413 return true; // Hamiltonian circuit found
414 }
415 }
416 } else {
417 // recursive step to try all vertices as the next one in the Hamiltonian
418 // circuit
419 for (const auto &neighbor : adj_list[v]) {
420 if (!visited[neighbor.first]) {
421 if (hamiltonian_circuit_util(neighbor.first,
422 visited,
423 path,
424 count + 1)) {
425 return true;
426 }
427 }
428 }
429 }
430
431 // backtrack
432 visited[v] = false;
433 path.pop_back();
434 return false;
435}

437int gpmp::Graph::eccentricity(int vertex) {
438 std::vector<int> distance(vertices, std::numeric_limits<int>::max());
439 distance[vertex] = 0;
440
441 std::queue<int> bfs_queue;
442 bfs_queue.push(vertex);
443
444 while (!bfs_queue.empty()) {
445 int v = bfs_queue.front();
446 bfs_queue.pop();
447
448 for (const auto &neighbor : adj_list[v]) {
449 int u = neighbor.first;
450 if (distance[u] == std::numeric_limits<int>::max()) {
451 distance[u] = distance[v] + 1;
452 bfs_queue.push(u);
453 }
454 }
455 }
456
457 // the eccentricity is the maximum distance from the given vertex
458 return *std::max_element(distance.begin(), distance.end());
459}

473int gpmp::Graph::chromatic_number() {
474 std::vector<int> result(vertices, -1);
475
476 // assign the first color to the first vertex
477 result[0] = 0;
478
479 // initialize available colors. For each vertex, remove colors used by its
480 // neighbors.
481 std::vector<bool> available(vertices, false);
482 for (const auto &neighbor : adj_list[0]) {
483 int v = neighbor.first;
484 if (result[v] != -1) {
485 available[result[v]] = true;
486 }
487 }
488
489 // assign colors to the remaining vertices
490 for (int v = 1; v < vertices; ++v) {
491 for (int c = 0; c < vertices; ++c) {
492 if (!available[c]) {
493 result[v] = c;
494 break;
495 }
496 }
497
498 // reset available colors for the next vertex
499 std::fill(available.begin(), available.end(), false);
500
501 // update available colors for neighbors
502 for (const auto &neighbor : adj_list[v]) {
503 int u = neighbor.first;
504 if (result[u] != -1) {
505 available[result[u]] = true;
506 }
507 }
508 }
509
510 // find the maximum color assigned, which represents the chromatic number
511 int chrom_num = *std::max_element(result.begin(), result.end()) + 1;
512
513 return chrom_num;
514}

562// method to find a maximum weight matching in the graph using greedy algorithm
563std::vector<std::pair<int, int>> gpmp::Graph::match_wt() {
564 std::vector<std::pair<int, int>> matching;
565
566 // sort edges in descending order based on weights
567 std::vector<std::pair<std::pair<int, int>, int>> wt_edges;
568 for (int v = 0; v < vertices; ++v) {
569 for (const auto &neighbor : adj_list[v]) {
570 int u = neighbor.first;
571 int weight = neighbor.second;
572 wt_edges.push_back({{v, u}, weight});
573 }
574 }
575
576 std::sort(wt_edges.rbegin(),
577 wt_edges.rend(),
578 [](const auto &a, const auto &b) { return a.second < b.second; });
579
580 // iterate through sorted edges and add to matching if both vertices are
581 // unmatched
582 for (const auto &edge : wt_edges) {
583 int v = edge.first.first;
584 int u = edge.first.second;
585 if (!is_matched[v] && !is_matched[u]) {
586 matching.push_back({v, u});
587 is_matched[v] = true;
588 is_matched[u] = true;
589 }
590 }
591
592 return matching;
593}

602// model
603void gpmp::Graph::planar_gen(int num_vertices, double radius) {
604 vertices = num_vertices;
605 adj_list.resize(vertices);
606
607 // generate random points in a 2D plane
608 std::vector<std::pair<double, double>> points;
609 for (int v = 0; v < vertices; ++v) {
610 double x = static_cast<double>(rand()) / RAND_MAX;
611 double y = static_cast<double>(rand()) / RAND_MAX;
612 points.push_back({x, y});
613 }
614
615 // connect vertices if their Euclidean distance is less than the specified
616 // radius
617 for (int v = 0; v < vertices; ++v) {
618 for (int u = v + 1; u < vertices; ++u) {
619 double distance = euclid_dist(points[v], points[u]);
620 if (distance < radius) {
621 // TODO the weight of the generated edges should probably be
622 // determined algorithmically
623 add_edge(v, u, 0);

629// method to check if the graph contains a subgraph that is a subdivision of K5
630bool gpmp::Graph::has_k5() {
631 for (int v = 0; v < vertices; ++v) {
632 if (adj_list[v].size() >= 5) {
633 std::set<int> neighbors;
634 for (const auto &neighbor : adj_list[v]) {
635 neighbors.insert(neighbor.first);
636 }
637
638 for (const auto &u : neighbors) {
639 std::set<int> common_neighbors;
640 for (const auto &neighbor : adj_list[u]) {
641 if (neighbors.find(neighbor.first) != neighbors.end()) {
642 common_neighbors.insert(neighbor.first);
643 }
644 }
645
646 for (const auto &w : common_neighbors) {
647 if (neighbors.find(w) != neighbors.end()) {
648 return true;
649 }
650 }
651 }
652 }
653 }
654
655 return false;
656}

692double gpmp::Graph::euclid_dist(const std::pair<double, double> &point1,

699std::vector<uint64_t> gpmp::Graph::compress() {
700 std::vector<uint64_t> compressed;
701 compressed.push_back(vertices);
702
703 for (int v = 0; v < vertices; ++v) {
704 // encode the number of neighbors using Elias Gamma encoding
705 std::bitset<64> binary_representation(adj_list[v].size() + 1);
706 int length = log2(adj_list[v].size() + 1) + 1;
707
708 // store the length of the encoded number
709 compressed.push_back(length);
710
711 // store the encoded number
712 compressed.push_back(
713 std::stoull(binary_representation.to_string(), 0, 2));
714
715 // store the neighbors
716 for (const auto &neighbor : adj_list[v]) {
717 compressed.push_back(neighbor.first);
718 }
719 }
720
721 return compressed;
722}

724// method to decompress a compressed graph using the Elias Gamma encoding
725void gpmp::Graph::decompress(const std::vector<uint64_t> &compressed) {
726 int index = 0;
727 vertices = compressed[index++];
728
729 adj_list.resize(vertices);
730
731 for (int v = 0; v < vertices; ++v) {
732 int length = compressed[index++];
733 uint64_t encoded_neighbors = compressed[index++];
734 std::bitset<64> binary_representation(encoded_neighbors);
735
736 // decode the number of neighbors
737 int num_neighbors =
738 std::stoi(binary_representation.to_string().substr(64 - length),
739 nullptr,
740 2) -
741 1;
742
743 // decode and add neighbors
744 for (int i = 0; i < num_neighbors; ++i) {
745 adj_list[v].emplace_back(compressed[index++],
746 0); // TODO assuming unweighted graph
747 }
748 }
