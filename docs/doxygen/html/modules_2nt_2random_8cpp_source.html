50uint64_t gpmp::core::rndm::LCG::operator()() {
51 // Ensure the state is non-negative
52 state &= 0x7FFFFFFFFFFFFFFFULL;
53
54 // Linear Congruential Generator algorithm
55 state = multiplier * state + increment;
56
57 // Overflow handling
58 if (state < increment) {
59 // Overflow occurred, adjust the state
60 state += increment;
61 }
62
63 return state;
64}

139uint32_t gpmp::core::rndm::pcg32(void) {
140 uint64_t x = __PCG_STATE;
141 unsigned count = (unsigned)(x >> 59); // 59 = 64 - 5
142
143 __PCG_STATE = x * __PCG_MULTPLR + __PCG_INCR;
144 x ^= x >> 18; // 18 = (64 - 27)/2
145 return rotr32((uint32_t)(x >> 27), count); // 27 = 32 - 5
146}
