50void gpmp::RedPike::encrypt(gpmp::RedPike::word *x,
51 const gpmp::RedPike::word *k) {
52 gpmp::RedPike::word rk0 = k[0];
53 gpmp::RedPike::word rk1 = k[1];
54
55 for (int i = 0; i < gpmp::RedPike::ROUNDS; i++) {
56 rk0 += gpmp::RedPike::CONST;
57 rk1 -= gpmp::RedPike::CONST;
58
59 x[0] ^= rk0;
60 x[0] += x[1];
61 x[0] = gpmp::RedPike::ROTL(x[0], x[1]);
62
63 x[1] = gpmp::RedPike::ROTR(x[1], x[0]);
64 x[1] -= x[0];
65 x[1] ^= rk1;
66 }
67
68 rk0 = x[0];
69 x[0] = x[1];
70 x[1] = rk0;
71}

73void gpmp::RedPike::decrypt(gpmp::RedPike::word *x,
74 const gpmp::RedPike::word *k) {
75 gpmp::RedPike::word dk[2] = {
76 k[1] - gpmp::RedPike::CONST * (gpmp::RedPike::ROUNDS + 1),
77 k[0] + gpmp::RedPike::CONST * (gpmp::RedPike::ROUNDS + 1)};
78
79 gpmp::RedPike::encrypt(x, dk);
80}
