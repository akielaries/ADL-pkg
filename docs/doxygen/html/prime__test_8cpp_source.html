60// Calculates (a * b) mod m
61uint64_t gpmp::PrimalityTest::mod_mul(uint64_t a, uint64_t b, uint64_t m) {
62 uint64_t res = 0;
63 while (b > 0) {
64 if (b & 1) {
65 res = (res + a) % m;
66 }
67 a = (2 * a) % m;
68 b >>= 1;
69 }
70 return res;
71}

118/*
119 * determining if a given number is likely to be prime
120 */
121bool gpmp::PrimalityTest::compute_miller_rabin(uint64_t d, uint64_t n) {
122 // Pick a random number in [2..n-2] Corner cases make sure that n
123 // > 4
124
125#ifdef __APPLE__
126 std::random_device rd;
127 std::mt19937 gen(rd());
128 std::uniform_int_distribution<uint64_t> dist(2, n - 3);
129 uint64_t a = dist(gen);
130#else
131 uint64_t a = 2 + rand() % (n - 4);
132#endif
133 /*
134 std::random_device rd;
135 std::mt19937_64 gen(rd());
136 std::uniform_int_distribution<uint64_t> dis(2, n - 4);
137 uint64_t a = dis(gen);
138 */
139
140 // Compute a^d % n
141 uint64_t x = mod_pow(a, d, n);
142
143 if (x == 1 || x == n - 1) {
144 return true;
145 }
146
147 // Keep squaring x while one of the following doesn't
148 // happen
149 // (I) d does not reach n-1
150 // (II) (x^2) % n is not 1
151 // (III) (x^2) % n is not n-1
152 while (d != n - 1) {
153 x = (x * x) % n;
154 d *= 2;
155
156 if (x == 1) {
157 return false;
158 }
159 if (x == n - 1) {
160 return true;
161 }
162 }
163
164 // Return composite
165 return false;
166}

168bool gpmp::PrimalityTest::witness(uint64_t n,
169 uint64_t d,
170 uint64_t a,
171 uint64_t s) {
172 uint64_t x = mod_pow(a, d, n);
173 if (x == 1 || x == n - 1) {
174 return false;
175 }
176 for (uint64_t r = 1; r < s; r++) {
177 x = mod_mul(x, x, n);
178 if (x == n - 1) {
179 return false;
180 }
181 }
182 return true;
183}

228bool gpmp::PrimalityTest::AKS(uint64_t n) {
229 // Check if n is a perfect power
230 for (uint64_t b = 2; b <= std::sqrt(n); b++) {
231 uint64_t a = std::round(std::pow(n, 1.0 / b));
232 // if (std::pow(a, b) == n) {
233 if (std::pow(a, b) < std::numeric_limits<double>::epsilon()) {
234 return false;
235 }
236 }
237
238 // Find the smallest r such that ord_r(n) > log2(n)^2
239 uint64_t r = 2;
240 while (r < n) {
241 bool is_coprime = true;
242 for (uint64_t i = 2; i <= std::sqrt(r); i++) {
243 if (r % i == 0 && n % i == 0) {
244 is_coprime = false;
245 break;
246 }
247 }
248 if (is_coprime) {
249 uint64_t t = 1;
250 for (uint64_t i = 0; i < std::floor(2 * std::log2(n)); i++) {
251 t = (t * r) % n;
252 }
253 if (t == 1) {
254 r++;
255 continue;
256 }
257 bool is_prime = true;
258 for (uint64_t i = 0; i < std::floor(std::log2(n)); i++) {
259 t = (t * t) % n;
260 if (t == 1) {
261 is_prime = false;
262 break;
263 }
264 }
265 if (is_prime) {
266 return true;
267 }
268 }
269 r++;
270 }
271 return false;
272}

321/*
322 * primality test determining if a number is composite or probably
323 * prime.
324 *
325 * a^(p-1)/2 = (a/p) (mod p)
326 */
327bool gpmp::PrimalityTest::solovoy_strassen(uint64_t p, uint64_t iters) {
328 if (p < 2)
329 return false;
330
331 if (p != 2 && p % 2 == 0)
332 return false;
333
334 for (uint64_t i = 0; i < iters; i++) {
335 // Generate a random number a
336 uint64_t a = rand() % (p - 1) + 1;
337 uint64_t jacobian = (p + jacobian_number(a, p)) % p;
338 uint64_t mod = mod_pow(a, (p - 1) / 2, p);
339
340 if (!jacobian || mod != jacobian)
341 return false;
342 }
343 return true;
344}

346/*
347 * Absolute Fermat Pseudoprimes referred to as Carmichael Numbers
348 * are composite integers satisfying the congruence forumla below
349 * b^n - 1 = b (mod n)
350 */
351bool gpmp::PrimalityTest::carmichael_num(uint64_t n) {
352 for (uint64_t b = 2; b < n; b++) {
353 // If "b" is relatively prime to n
354 if (ba.op_gcd(b, n) == 1)
355 // And pow(b, n-1)%n is not 1, return false.
356 if (mod_pow(b, n - 1, n) != 1)
357 return false;
358 }
359 return true;
360}

396/*
397 * Euler's Totient Function counts the positive numbers up until a
398 * given integer
399 *
400 * Definition: For any positive integer n, Î¦(n) is the number of all
401 * positive integers less than or equal to n that are relatively prime
402 * to n.
403 *
404 * Example: Î¦(1) = 1
405 * Î¦(2) = 1
406 * Î¦(3) = 2
407 * Î¦(4) = 2
408 * Î¦(5) = 4
409 */
410uint64_t gpmp::PrimalityTest::ETF(uint64_t n) {
411 uint64_t result = 1;
412
413 for (uint64_t index = 2; uint64_t(index) < n; index++) {
414 if (ba.op_gcd(index, n) == 1) {
415 result++;
416 }
417 }
418
