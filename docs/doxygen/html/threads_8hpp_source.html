60class ThreadPool {
61 private:
62 // VECTOR of threads to execute tasks
63 std::vector<std::thread> workers;
64 // QUEUE of tasks to be executed
65 std::queue<std::function<void()>> tasks;
66 // MUTEX synchronizing access to the QUEUE of tasks
67 std::mutex queue_mutex;
68 // CONDITIONAL to notify waiting threads when queue gets populated
69 std::condition_variable condition;
70 // BOOL indicating if ThreadPool should stop execution
71 bool stop;
72
73 public:
78 ThreadPool() : ThreadPool(std::thread::hardware_concurrency()) {
79 }
80
88 explicit ThreadPool(int numThreads) : stop(false) {
89
90 // traverse through the number of threads specified
91 for (int i = 0; i < numThreads; ++i) {
92 // add a new thread to the vector storing workers using lambda
93 // function
94 workers.emplace_back([this] {
95 for (;;) {
96 // worker thread creates task object that holds next task to
97 // be executed
98 std::function<void()> task_obj;
99
100 // this "symbolizes" the critical section of the TheadPool
101 // class
102 {
103 // worker thread locks queue_mutex
104 std::unique_lock<std::mutex> lock(this->queue_mutex);
105 // wait on conditional_variable (ThreadPool stop OR
106 // queued task), wait() locks/unlocks based on condition
107 // result
108 this->condition.wait(lock, [this] {
109 return this->stop || !this->tasks.empty();
110 });
111 // based on stop OR awaiting tasks, return from the
112 // thread
113 if (this->stop && this->tasks.empty()) {
114 return;
115 }
116
117 // if above isnt met, move first task in TASKS queue to
118 // the task object to transfer ownership
119 task_obj = std::move(this->tasks.front());
120
121 // pop the handed off task to make room for a new one.
122 // only ONE thread should remove a task from the queue
123 // at a time
124 this->tasks.pop();
125 }
126
127 // EXECUTE THE HANDED OFF TASK
128 task_obj();
129 }
130 });
131 }
132 }

147 template <class F, class... Args>
148 auto enqueue(F &&f, Args &&...args)
149 -> std::future<typename std::result_of<F(Args...)>::type> {
150
151 // this is the return type of the passed in function
152 using return_type = typename std::result_of<F(Args...)>::type;
153 // * SHARED POINTER to PACKAGED TASK used to store the passed in i
154 // function + its arguments
155 // * std::bind used to create function object binded to the
156 // function `f` + its args to the packaged tasks
157 // * std::forward used for forwarding an argument to another
158 // function
159 auto task = std::make_shared<std::packaged_task<return_type()>>(
160 std::bind(std::forward<F>(f), std::forward<Args>(args)...));
161
162 // the FUTURE obj retrieves the return value of the function passed in
163 std::future<return_type> res = task->get_future();
164 {
165 // aquire lock on queue_mutex for synchronization
166 std::unique_lock<std::mutex> lock(queue_mutex);
167 // check if threadpool stop is initiated
168 if (stop) {
169 throw std::runtime_error("enqueue on stopped ThreadPool");
170 }
171 // add a task using emplace to the queue as a lambda that calls the
172 // packaged task
173 tasks.emplace([task]() { (*task)(); });
174 } // once this is hit, unique_lock is out of scope & mutex is
175 // automatically unlocked
176 // notify one waiting thread of one new task added to the queue
177 condition.notify_one();
178 // the return is the future object
179 return res;
180 }
181
182 ~ThreadPool() {
183 {
184 // lock queue_mutex & set stop to true
185 std::unique_lock<std::mutex> lock(queue_mutex);
186 stop = true;
187 }
188 // unblock all threads
189 condition.notify_all();
190 // treaverse threads and join
191 for (std::thread &worker : workers) {
192 worker.join();
193 }
194 }
195};

242static void parallel_for(unsigned nb_elements,
243 std::function<void(int start, int end)> functor,
244 bool use_threads = true) {
245 // -------
246 unsigned nb_threads_hint = 10; // std::thread::hardware_concurrency();
247 unsigned nb_threads = nb_threads_hint == 0 ? 8 : (nb_threads_hint);
248
249 unsigned batch_size = nb_elements / nb_threads;
250 unsigned batch_remainder = nb_elements % nb_threads;
251
252 std::vector<std::thread> my_threads(nb_threads);
253
254 if (use_threads) {
255 // Multithread execution
256 for (unsigned i = 0; i < nb_threads; ++i) {
257 int start = i * batch_size;
258 my_threads[i] = std::thread(functor, start, start + batch_size);
259 }
260 } else {
261 // Single thread execution (for easy debugging)
262 for (unsigned i = 0; i < nb_threads; ++i) {
263 int start = i * batch_size;
264 functor(start, start + batch_size);
265 }
266 }
267
268 // Deform the elements left
269 int start = nb_threads * batch_size;
270 functor(start, start + batch_remainder);
271
272 // Wait for the other thread to finish their task
273 if (use_threads) {
274 std::for_each(my_threads.begin(),
275 my_threads.end(),
276 std::mem_fn(&std::thread::join));
277 }
278}
