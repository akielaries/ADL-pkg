54 void dijkstra(const std::vector<std::vector<Edge>> &graph,
55  int source,
56  std::vector<int> &distances,
57  std::vector<int> &parents) {
58  int n = graph.size();
59  distances.assign(n, INF);
60  parents.assign(n, -1);
61 
62  distances[source] = 0;
63  std::priority_queue<std::pair<int, int>,
64  std::vector<std::pair<int, int>>,
65  std::greater<std::pair<int, int>>>
66  pq;
67  pq.push(std::make_pair(0, source));
68 
69  while (!pq.empty()) {
70  int current = pq.top().second;
71  int distance = pq.top().first;
72  pq.pop();
73 
74  if (distance > distances[current])
75  continue;
76 
77  for (const Edge &edge : graph[current]) {
78  int neighbor = edge.to;
79  int new_distance = distance + edge.weight;
80 
81  if (new_distance < distances[neighbor]) {
82  distances[neighbor] = new_distance;
83  parents[neighbor] = current;
84  pq.push(std::make_pair(new_distance, neighbor));
85  }
86  }
87  }
88 }
