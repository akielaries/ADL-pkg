341 template<class T, swig::AssignmentType A>
342 SWIGINTERN void SWIG_assign(SwigClassWrapper* self, SwigClassWrapper other) {
343  typedef swig::AssignmentPolicy<T, A> Policy_t;
344 
345  if (self->cptr == NULL) {
346  /* LHS is unassigned */
347  if (other.cmemflags & SWIG_MEM_RVALUE) {
348  /* Capture pointer from RHS, clear 'moving' flag */
349  *self = swig::capture(other);
350  } else {
351  /* Aliasing another class; clear ownership or copy smart pointer */
352  *self = Policy_t::alias(other);
353  }
354  } else if (other.cptr == NULL) {
355  /* Replace LHS with a null pointer */
356  *self = Policy_t::destroy(*self);
357  } else if (self->cptr == other.cptr) {
358  /* Self-assignment: ignore */
359  } else if (other.cmemflags & SWIG_MEM_RVALUE) {
360  /* Transferred ownership from a variable that's about to be lost.
361  * Move-assign and delete the transient data */
362  *self = Policy_t::move_alias(*self, other);
363  } else {
364  /* RHS shouldn't be deleted, alias to LHS */
365  *self = Policy_t::copy_alias(*self, other);
366  }
367 }
