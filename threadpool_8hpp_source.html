79  explicit ThreadPool(int numThreads) : stop(false) {
80 
81  // traverse through the number of threads specified
82  for (int i = 0; i < numThreads; ++i) {
83  // add a new thread to the vector storing workers using lambda
84  // function
85  workers.emplace_back([this] {
86  for (;;) {
87  // worker thread creates task object that holds next task to
88  // be executed
89  std::function<void()> task_obj;
90 
91  // this "symbolizes" the critical section of the TheadPool
92  // class
93  {
94  // worker thread locks queue_mutex
95  std::unique_lock<std::mutex> lock(this->queue_mutex);
96  // wait on conditional_variable (ThreadPool stop OR
97  // queued task), wait() locks/unlocks based on condition
98  // result
99  this->condition.wait(lock, [this] {
100  return this->stop || !this->tasks.empty();
101  });
102  // based on stop OR awaiting tasks, return from the
103  // thread
104  if (this->stop && this->tasks.empty()) {
105  return;
106  }
107 
108  // if above isnt met, move first task in TASKS queue to
109  // the task object to transfer ownership
110  task_obj = std::move(this->tasks.front());
111 
112  // pop the handed off task to make room for a new one.
113  // only ONE thread should remove a task from the queue
114  // at a time
115  this->tasks.pop();
116  }
117 
118  // EXECUTE THE HANDED OFF TASK
119  task_obj();
120  }
121  });
122  }
123  }

138  template <class F, class... Args>
139  auto enqueue(F &&f, Args &&... args)
140  -> std::future<typename std::result_of<F(Args...)>::type> {
141 
142  // this is the return type of the passed in function
143  using return_type = typename std::result_of<F(Args...)>::type;
144  // * SHARED POINTER to PACKAGED TASK used to store the passed in i
145  // function + its arguments
146  // * std::bind used to create function object binded to the
147  // function `f` + its args to the packaged tasks
148  // * std::forward used for forwarding an argument to another
149  // function
150  auto task = std::make_shared<std::packaged_task<return_type()>>(
151  std::bind(std::forward<F>(f), std::forward<Args>(args)...));
152 
153  // the FUTURE obj retrieves the return value of the function passed in
154  std::future<return_type> res = task->get_future();
155  {
156  // aquire lock on queue_mutex for synchronization
157  std::unique_lock<std::mutex> lock(queue_mutex);
158  // check if threadpool stop is initiated
159  if (stop) {
160  throw std::runtime_error("enqueue on stopped ThreadPool");
161  }
162  // add a task using emplace to the queue as a lambda that calls the
163  // packaged task
164  tasks.emplace([task]() { (*task)(); });
165  }
166  // notify one waiting thread of one new task added to the queue
167  condition.notify_one();
168  // the return is the future object
169  return res;
170  }
171 
172  ~ThreadPool() {
173  {
174  // lock queue_mutex & set stop to true
175  std::unique_lock<std::mutex> lock(queue_mutex);
176  stop = true;
177  }
178  // unblock all threads
179  condition.notify_all();
180  // treaverse threads and join
181  for (std::thread &worker : workers) {
182  worker.join();
183  }
184  }
185 };
