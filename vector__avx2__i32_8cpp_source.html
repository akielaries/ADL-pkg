62 template <typename T>
63 void gpmp::linalg::vector_add_i32(const T *data1,
64  const T *data2,
65  T *result_data,
66  size_t size) {
67  if (size > 16) {
68  size_t i = 0;
69  for (; i < size - 7; i += 8) {
70  // Load 8 elements from vec1 and vec2
71  __m256i a = _mm256_loadu_si256(
72  reinterpret_cast<const __m256i *>(data1 + i));
73  __m256i b = _mm256_loadu_si256(
74  reinterpret_cast<const __m256i *>(data2 + i));
75 
76  // Perform vectorized addition
77  __m256i c = _mm256_add_epi32(a, b);
78 
79  // Store the result back to result vector
80  _mm256_storeu_si256(reinterpret_cast<__m256i *>(result_data + i),
81  c);
82  }
83  for (; i < size; ++i) {
84  result_data[i] = data1[i] + data2[i];
85  }
86  }
87 
88  else {
89  // if size is not a multiple of 8, perform standard addition
90  for (size_t i = 0; i < size; ++i) {
91  result_data[i] = data1[i] + data2[i];
92  }
93  }
94 }

166 void gpmp::linalg::scalar_mult_i32(const T *data,
167  int scalar,
168  T *result_data,
169  size_t size) {
170  if (size > 16) {
171  size_t i = 0;
172  __m256i scalar_vec = _mm256_set1_epi32(scalar);
173  // Perform vectorized multiplication as long as there are at least 8
174  // elements remaining
175  for (; i < size - 7; i += 8) {
176  // Load 8 elements from vec
177  __m256i a =
178  _mm256_loadu_si256(reinterpret_cast<const __m256i *>(data + i));
179 
180  // Perform vectorized multiplication
181  __m256i c = _mm256_mullo_epi32(a, scalar_vec);
182 
183  // Store the result back to result vector
184  _mm256_storeu_si256(reinterpret_cast<__m256i *>(result_data + i),
185  c);
186  }
187 
188  // Perform standard multiplication on the remaining elements
189  for (; i < size; ++i) {
190  result_data[i] = data[i] * scalar;
191  }
192  } else {
193  for (size_t i = 0; i < size; ++i) {
194  result_data[i] = data[i] * scalar;
195  }
196  }

214 template <typename T>
215 T gpmp::linalg::dot_product_i32(const T *vec1, const T *vec2, size_t size) {
216  int result = 0;
217  if (size > 16) {
218  size_t i = 0;
219  // Perform vectorized multiplication and addition as long as there are
220  // at least 8 elements remaining
221  for (; i < size - 7; i += 8) {
222  // Load 8 elements from vec1 and vec2
223  __m256i a =
224  _mm256_loadu_si256(reinterpret_cast<const __m256i *>(vec1 + i));
225  __m256i b =
226  _mm256_loadu_si256(reinterpret_cast<const __m256i *>(vec2 + i));
227 
228  // Perform vectorized multiplication and addition
229  __m256i c = _mm256_mullo_epi32(a, b);
230  __m256i sum = _mm256_hadd_epi32(c, c);
231  sum = _mm256_hadd_epi32(sum, sum);
232 
233  // Accumulate the result
234  result += _mm256_extract_epi32(sum, 0);
235  result += _mm256_extract_epi32(sum, 4);
236  }
237 
238  // Perform standard dot product on the remaining elements
239  for (; i < size; ++i) {
240  result += vec1[i] * vec2[i];
241  }
242  } else {
243  for (size_t i = 0; i < size; ++i) {
244  result += vec1[i] * vec2[i];
245  }
246  }
247 
248  return result;
249 }
