47  const uint32_t P = 0xB7E15163;
48  const uint32_t Q = 0x9E3779B9;
49  S[0] = P;

51  S[i] = S[i - 1] + Q;

92  uint32_t A = ciphertext[0], B = ciphertext[1];
93  for (int i = r; i >= 1; --i) {
94  B = rotr(B - S[2 * i + 1], A) ^ A;

111  uint32_t block[2];
