116  bool display_all = false) {
117  int num_columns = data.first.size();
118  int num_rows = data.second.size();

120 
121  for (const auto &header : data.first) {
122  std::cout << header << "\t";
123  }
124  std::cout << std::endl;
125 
126  int num_elements = data.second.size();
127  if (!display_all && num_elements > MAX_ROWS) {
128  for (int i = 0; i < SHOW_ROWS; i++) {
129  for (const auto &value : data.second[i]) {
130  std::cout << value << "\t";
131  }
132  std::cout << std::endl;
133  }
134  num_omitted_rows = num_elements - MAX_ROWS;
135  std::cout << "...\n";
136  std::cout << "[" << num_omitted_rows << " rows omitted]\n";
137  for (int i = num_elements - SHOW_ROWS; i < num_elements; i++) {
138  for (const auto &value : data.second[i]) {
139  std::cout << value << "\t";
140  }
141  std::cout << std::endl;
142  }
143  } else {
144  for (const auto &row : data.second) {
145  for (const auto &value : row) {

150  }
