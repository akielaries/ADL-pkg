171 // micro kernel that multiplies panels from A and B
172 void gpmp::linalg::DGEMM::dgemm_micro_kernel(int kc,
173  double alpha,
174  const double *A,
175  const double *B,
176  double beta,
177  double *C,
178  int incRowC,
179  int incColC) {
180  double AB[BLOCK_SZ_MR * BLOCK_SZ_NR];
181 
182  int i, j, l;
183 
184  // Compute AB = A*B
185  for (l = 0; l < BLOCK_SZ_MR * BLOCK_SZ_NR; ++l) {
186  AB[l] = 0;
187  }
188  for (l = 0; l < kc; ++l) {
189  for (j = 0; j < BLOCK_SZ_NR; ++j) {
190  for (i = 0; i < BLOCK_SZ_MR; ++i) {
191  AB[i + j * BLOCK_SZ_MR] += A[i] * B[j];
192  }
193  }
194  A += BLOCK_SZ_MR;
195  B += BLOCK_SZ_NR;
196  }
197 
198  // Update C <- beta*C
199  if (fabs(beta - 0.0) < std::numeric_limits<double>::epsilon()) {
200  for (j = 0; j < BLOCK_SZ_NR; ++j) {
201  for (i = 0; i < BLOCK_SZ_MR; ++i) {
202  C[i * incRowC + j * incColC] = 0.0;
203  }
204  }
205  } else if (fabs(beta - 1.0) > std::numeric_limits<double>::epsilon()) {
206  for (j = 0; j < BLOCK_SZ_NR; ++j) {
207  for (i = 0; i < BLOCK_SZ_MR; ++i) {
208  C[i * incRowC + j * incColC] *= beta;
209  }
210  }
211  }
212 
213  // Update C <- C + alpha*AB (note: the case alpha==0.0 was already treated
214  // in
215  // the above layer dgemm_nn)
216  if (fabs(alpha - 1.0) < std::numeric_limits<double>::epsilon()) {
217  for (j = 0; j < BLOCK_SZ_NR; ++j) {
218  for (i = 0; i < BLOCK_SZ_MR; ++i) {
219  C[i * incRowC + j * incColC] += AB[i + j * BLOCK_SZ_MR];
220  }
221  }
222  }
223 
224  else {
225  for (j = 0; j < BLOCK_SZ_NR; ++j) {
226  for (i = 0; i < BLOCK_SZ_MR; ++i) {
227  C[i * incRowC + j * incColC] += alpha * AB[i + j * BLOCK_SZ_MR];
228  }
229  }
230  }
231 }
