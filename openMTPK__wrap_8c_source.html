543  Note: if start == end at the beginning of the function, we go all the way around

546 SWIGRUNTIME swig_type_info *
547 SWIG_MangledTypeQueryModule(swig_module_info *start,
548  swig_module_info *end,
549  const char *name) {
550  swig_module_info *iter = start;
551  do {
552  if (iter->size) {
553  size_t l = 0;
554  size_t r = iter->size - 1;
555  do {
556  /* since l+r >= 0, we can (>> 1) instead (/ 2) */
557  size_t i = (l + r) >> 1;
558  const char *iname = iter->types[i]->name;
559  if (iname) {
560  int compare = strcmp(name, iname);
561  if (compare == 0) {
562  return iter->types[i];
563  } else if (compare < 0) {
564  if (i) {
565  r = i - 1;
566  } else {
567  break;
568  }
569  } else if (compare > 0) {
570  l = i + 1;
571  }
572  } else {
573  break; /* should never happen */
574  }
575  } while (l <= r);
576  }
577  iter = iter->next;
578  } while (iter != end);

967  SWIG_Cast (void *source, swig_type_info *source_type,
968  void **ptr, swig_type_info *dest_type)
969  {
970  if( !source ) { /* Special case for NULL. This is a popular question
971  for other modules on the list, so I want an easy way out... */
972  *ptr = 0;
973  return 0;
974  }

978  source_type ? source_type->str : "<none>",
979  dest_type ? dest_type->str : "<none>" );
980 #endif
981  if (dest_type != source_type) {
982  /* We have a type mismatch. Will have to look through our type
983  mapping table to figure out whether or not we can accept this
984  datatype.
985  --
986  Ignore typechecks for void *. Allow any conversion. */
987  if( !dest_type || !source_type ||
988  !strcmp(dest_type->name,"_p_void") ||
989  !strcmp(source_type->name,"_p_void") ) {
990  *ptr = source;
991  return 0;
992  } else {
993  swig_cast_info *tc =
994  SWIG_TypeCheckStruct(source_type, dest_type );
995 #ifdef TYPE_CAST_VERBOSE
996  fprintf( stderr, "Typecheck -> %s\n",
997  tc ? tc->type->str : "<none>" );
998 #endif
999  if( tc ) {
1000  int newmemory = 0;
1001  *ptr = SWIG_TypeCast(tc, source, &newmemory);
1002  assert(!newmemory); /* newmemory handling not yet implemented */
1003  return 0;
1004  } else
1005  return -1;
1006  }
1007  } else {
1008  *ptr = source;
1009  return 0;

2635  * swig_cast_info structures and swig_cast_info structures store pointers back
2636  * to swig_type_info structures, we need some lookup code at initialization.
2637  * The idea is that swig generates all the structures that are needed.
2638  * The runtime then collects these partially filled structures.
2639  * The SWIG_InitializeModule function takes these initial arrays out of
2640  * swig_module, and does all the lookup, filling in the swig_module.types
2641  * array with the correct data and linking the correct swig_cast_info
2642  * structures together.
2643  *
2644  * The generated swig_type_info structures are assigned statically to an initial
2645  * array. We just loop through that array, and handle each type individually.
2646  * First we lookup if this type has been already loaded, and if so, use the
2647  * loaded structure instead of the generated one. Then we have to fill in the
2648  * cast linked list. The cast data is initially stored in something like a
2649  * two-dimensional array. Each row corresponds to a type (there are the same
2650  * number of rows as there are in the swig_type_initial array). Each entry in
2651  * a column is one of the swig_cast_info structures for that type.
2652  * The cast_initial array is actually an array of arrays, because each row has
2653  * a variable number of columns. So to actually build the cast linked list,
2654  * we find the array of casts associated with the type, and loop through it
2655  * adding the casts to the list. The one last trick we need to do is making
2656  * sure the type pointer in the swig_cast_info struct is correct.
2657  *
2658  * First off, we lookup the cast->type name to see if it is already loaded.
2659  * There are three cases to handle:
2660  * 1) If the cast->type has already been loaded AND the type we are adding
2661  * casting info to has not been loaded (it is in this module), THEN we
2662  * replace the cast->type pointer with the type pointer that has already
2663  * been loaded.

2665  * cast->type) are loaded, THEN the cast info has already been loaded by
2666  * the previous module so we just ignore it.
