75 // function to compute the projection of one vector onto another
76 void gpmp::linalg::std_vector_projection(const std::vector<double> &vec,
77  const std::vector<double> &onto_vec,
78  std::vector<double> &result) {

80  double onto_vec_norm_sq = std_dot_product(onto_vec, onto_vec);

84  result[i] = (dot / onto_vec_norm_sq) * onto_vec[i];
