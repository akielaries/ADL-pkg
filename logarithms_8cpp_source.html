58 int64_t mtpk::Logarithms::pollard_rho_log(int g, int y, int p) {
59  int64_t x = rand() % (p - 1) + 1;
60 
61 #ifdef USE_SRAND
62  srand(time(NULL));
63  x = rand() % (p - 1) + 1;
64 #endif
65 
66  int64_t y1 = y;
67  int64_t y2 = y;
68  int64_t d = 1;
69  std::unordered_map<int64_t, int64_t> values;
70 
71  // Compute the iterated values of the sequence x_n = g^x_n-1 mod p
72  // until a collision is found
73  while (d == 1) {
74  x = (x * x + 1) % p;
75  y1 = (y1 * g) % p;
76  y2 = (y2 * g) % p;
77  y2 = (y2 * g) % p;
78  d = std::__gcd(abs(y2 - y1), p);
79  values[x] = y1;
80  }
81 
82  // Compute the discrete logarithm using the collision point
83  if (d == p) {
84  return -1;
85  } else {
86  int64_t x0 = abs(y2 - y1) / d;
87  int64_t y0 = values[x0];
88  int64_t k = 1;
89  while (y0 != y) {
90  y0 = (y0 * g) % p;
91  k++;
92  }
93  return k * x0 % (p - 1);
94  }
95 }

98 int64_t mtpk::Logarithms::BSGS(int64_t a, int64_t b, int64_t m) {
99  int64_t n = (int64_t)sqrt(m) + 1;
100 
101  std::unordered_map<int64_t, int64_t> value;
102 
103  // store all values of a^(n*i) of LHS
104  for (int64_t i = n; i >= 1; --i) {
105  value[__LOG_PRIMES__.mod_pow(a, i * n, m)] = i;
106  }
107 
108  for (int64_t j = 0; j < n; ++j) {
109  // calculate (a ^ j) * b and check
110  int64_t cur = (__LOG_PRIMES__.mod_pow(a, j, m) * b) % m;
111 
112  // If collision occurs i.e., LHS = RHS
113  if (value[cur]) {
114  int64_t ans = value[cur] * n - j;
115  // Check whether ans lies below m or not
116  if (ans < m) {
117  return ans;
118  }
119  }
120  }
121  return -1;
122 }
