189  explicit SecondaryMLP(std::vector<size_t> layer_units,
190  long double lr = .001)
191  : layer_units(layer_units), wt_mtx(), bias_vectors(), lr(lr) {
192  // traverse the elements
193  for (size_t i = 0; i < layer_units.size() - 1; ++i) {
194  // size of inputs
195  size_t inputs{layer_units[i]};
196  // size of outputs
197  size_t outputs{layer_units[i + 1]};
198 
199  // set to random Guassian Noise related values
200  // weights
201  auto gauss_wt = mtpk::mtx<T>::randn(outputs, inputs);
202  wt_mtx.push_back(gauss_wt);
203  // biases
204  auto bias_wt = mtpk::mtx<T>::randn(outputs, 1);
205  bias_vectors.push_back(bias_wt);
206  // activation function
207  activations.resize(layer_units.size());
208  }
209  }
216  auto prop_forwards(Matrix<T> x) {
217  assert(get<0>(x.shape) == layer_units[0] && get<1>(x.shape));
218  // input = to previously declared acitvations method
219  activations[0] = x;
220  Matrix prev(x);
221 
222  // traverse layer units
223  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
224  Matrix y = wt_mtx[i].mult(prev);
225  y = y + bias_vectors[i];
226  y = y.apply_function(sigmoid_activ);
227  activations[i + 1] = y;
228  prev = y;
229  }
230  return prev;
231  }
240  void prop_backwards(Matrix<T> target) {
241  assert(get<0>(target.shape) == layer_units.back());
242  // calculate the error, target - ouput
243  auto y = target;
244  auto y_hat = activations.back();
245  auto err = target - y_hat;
246  // back propagate the error calculated from output to input
247  // and step the weights
248  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
249  // calculate errors for previous layer
250  auto wt = wt_mtx[i].T();
251  auto prior_errs = wt.mult(err);
252  auto outputs_d =
253  activations[i + 1].apply_function(sigmoid_deriv);
254  auto gradients = err.hadamard(outputs_d);
255  gradients = gradients.scalar_mult(lr);
256  auto trans_a = activations[i].T();
257  auto gradients_wt = gradients.mult(trans_a);
258 
259  // adjust the networks weights based on propagation
260  // technique
261  bias_vectors[i] = bias_vectors[i].add(gradients);
262  wt_mtx[i] = wt_mtx[i].add(gradients_wt);
263  err = prior_errs;
264  }
265  }
