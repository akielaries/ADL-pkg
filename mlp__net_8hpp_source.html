202  inline long double sigmoid_activ(long double x) {
203  return 1.0f / (1 + exp(-x));

211  inline long double sigmoid_deriv(long double x) {
212  return (x * (1 - x));
213  }

216  std::vector<Matrix<T>> bias_vectors;
217  std::vector<Matrix<T>> wt_mtx;
218  std::vector<Matrix<T>> activations;

227  explicit SecondaryMLP(std::vector<size_t> layer_units,
228  long double lr = .001)
229  : layer_units(layer_units), wt_mtx(), bias_vectors(), lr(lr) {
230  // traverse the elements
231  for (size_t i = 0; i < layer_units.size() - 1; ++i) {
232  // size of inputs
233  size_t inputs{layer_units[i]};
234  // size of outputs
235  size_t outputs{layer_units[i + 1]};
236 
237  // set to random Guassian Noise related values
238  // weights
239  auto gauss_wt = mtpk::mtx<T>::randn(outputs, inputs);
240  wt_mtx.push_back(gauss_wt);
241  // biases
242  auto bias_wt = mtpk::mtx<T>::randn(outputs, 1);
243  bias_vectors.push_back(bias_wt);
244  // activation function
245  activations.resize(layer_units.size());
246  }
247  }
254  auto prop_forwards(Matrix<T> x) {
255  assert(get<0>(x.shape) == layer_units[0] && get<1>(x.shape));
256  // input = to previously declared acitvations method
257  activations[0] = x;
258  Matrix prev(x);
259 
260  // traverse layer units
261  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
262  Matrix y = wt_mtx[i].mult(prev);
263  y = y + bias_vectors[i];
264  y = y.apply_function(sigmoid_activ);
265  activations[i + 1] = y;
266  prev = y;
267  }
268  return prev;
269  }
278  void prop_backwards(Matrix<T> target) {
279  assert(get<0>(target.shape) == layer_units.back());
280  // calculate the error, target - ouput
281  auto y = target;
282  auto y_hat = activations.back();
283  auto err = target - y_hat;
284  // back propagate the error calculated from output to input
285  // and step the weights
286  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
287  // calculate errors for previous layer
288  auto wt = wt_mtx[i].T();
289  auto prior_errs = wt.mult(err);
290  auto outputs_d =
291  activations[i + 1].apply_function(sigmoid_deriv);
292  auto gradients = err.hadamard(outputs_d);
293  gradients = gradients.scalar_mult(lr);
294  auto trans_a = activations[i].T();
295  auto gradients_wt = gradients.mult(trans_a);
296 
297  // adjust the networks weights based on propagation
298  // technique
299  bias_vectors[i] = bias_vectors[i].add(gradients);
300  wt_mtx[i] = wt_mtx[i].add(gradients_wt);
301  err = prior_errs;
302  }
303  }
304 };
