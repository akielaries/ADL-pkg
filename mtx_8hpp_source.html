489  void std_mtx_add(const T *A, const T *B, T *C, int rows, int cols) {
490  // MTX A AND B MUST BE SAME SIZE
491  for (int i = 0; i < rows; ++i) {
492  for (int j = 0; j < cols; ++j) {
493  // perform matrix addition
494  C[i * cols + j] = A[i * cols + j] + B[i * cols + j];
495  }
496  }

528  U sum = 0; // Use T type for sum
