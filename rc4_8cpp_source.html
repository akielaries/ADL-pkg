69 void mtpk::RC4::KSA(char *key, unsigned char *S, int swap_type) {
70  uint32_t len = strlen(key);
71  int j = 0;
72 
73  for (int i = 0; i < BYTE_LIMIT; i++) {
74  S[i] = i;
75  }
76 
77  for (int i = 0; i < BYTE_LIMIT; i++) {
78  j = (j + S[i] + key[i % len]) & BITS;
79 
80  // choose swap algorithm based off swap_type
81  if (swap_type == 0) {
82  XOR_swap(&S[i], &S[j]);
83  } else if (swap_type == 1) {
84  trad_swap(&S[i], &S[j]);
85  } else if (swap_type == 2) {
86  byte_swap(&S[i], &S[j]);
87  }
88  }
89 }

91 void mtpk::RC4::PRGA(unsigned char *S, char *plaintext,
92  unsigned char *ciphertext, int swap_type) {
93  int i = 0;
94  int j = 0;
95 
96  for (size_t n = 0, len = strlen(plaintext); n < len; n++) {
97  i = (i + 1) & BITS;
98  j = (j + S[i]) & BITS;
99 
100  // choose swap algorithm based off swap_type
101  if (swap_type == 0) {
102  XOR_swap(&S[i], &S[j]);
103  } else if (swap_type == 1) {
104  trad_swap(&S[i], &S[j]);
105  } else if (swap_type == 2) {
106  byte_swap(&S[i], &S[j]);
107  }
108 
109  uint32_t rnd = S[(S[i] + S[j]) & BITS];
110 
111  ciphertext[n] = rnd ^ plaintext[n];
112  }
113 }
