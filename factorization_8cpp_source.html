53 /*
54  * algorithm for integer factorization proportial to the runtime of
55  * the square root of the size of the smallest prime factor of the
56  * composite factor being factorized. Given a positive and composite
57  * integer n, find a divisor of it.
58  *
59  * x_n-1 = x^2_n + a(mod n)
60  *
61  * n = 1111, set x_0 = 2 & f(x) = x^2 + 1
62  *
63  * x_1 = 5
64  * x_2 = 26 gcd(26 - 5, 1111) = 1
65  * x_3 = 677
66  * x_4 = 598 gcd(698 - 26, 1111) = 11
67  */
68 int64_t mtpk::Factorization::pollard_rho(int64_t n) {
69  /* initialize random seed */
70  srand(time(NULL));
71 
72  /* no prime divisor for 1 */
73  if (n == 1)
74  return n;
75 
76  /* even number means one of the divisors is 2 */
77  if (n % 2 == 0)
78  return 2;
79 
80  /* we will pick from the range [2, N) */
81  int64_t x = (rand() % (n - 2)) + 2;
82  int64_t y = x;
83 
84  /* the constant in f(x).
85  * Algorithm can be re-run with a different c
86  * if it throws failure for a composite. */
87  int64_t c = (rand() % (n - 1)) + 1;
88 
89  /* Initialize candidate divisor (or result) */
90  int64_t divisor = 1;
91 
92  /* until the prime factor isn't obtained.
93  If n is prime, return n */
94  while (divisor == 1) {
95  /* Tortoise Move: x(i+1) = f(x(i)) */
96  // take power
97  // calculate modulus
98  x = (mod_pow(x, 2, n) + c + n) % n;
99 
100  /* Hare Move: y(i+1) = f(f(y(i))) */
101  y = (mod_pow(y, 2, n) + c + n) % n;
102  y = (mod_pow(y, 2, n) + c + n) % n;
103 
104  /* check gcd of |x-y| and n */
105  divisor = ba.op_gcd(abs(x - y), n);
106  // divisor = std::__gcd(abs(x - y), n);
107  /* retry if the algorithm fails to find prime factor
108  * with chosen x and c */
109  if (divisor == n)
110  return pollard_rho(n);
111  }
112 
113  return divisor;
114 }
