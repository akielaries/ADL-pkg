82 int64_t mtpk::Factorization::pollard_rho(int64_t n) {
83  /* initialize random seed */
84  std::mt19937_64 rng(
85  std::chrono::steady_clock::now().time_since_epoch().count());
86 
87  /* no prime divisor for 1 */
88  if (n == 1) {
89  return n;
90  }
91 
92  /* even number means one of the divisors is 2 */
93  if (n % 2 == 0) {
94  return 2;
95  }
96 
97  /* we will pick from the range [2, N) */
98  std::uniform_int_distribution<int64_t> unif_dist(2, n - 1);
99  int64_t x = unif_dist(rng);
100  int64_t y = x;
101 
102  /* the constant in f(x).
103  * Algorithm can be re-run with a different c
104  * if it throws failure for a composite. */
105  std::uniform_int_distribution<int64_t> c_dist(1, n - 1);
106  int64_t c = c_dist(rng);
107 
108  /* Initialize candidate divisor (or result) */
109  int64_t divisor = 1;
110 
111  /* until the prime factor isn't obtained.
112  If n is prime, return n */
113  while (divisor == 1) {
114  /* Tortoise Move: x(i+1) = f(x(i)) */
115  x = __FACT_PRIMES__.mod_pow(x, 2, n) + c;
116  if (x >= n) {
117  x -= n;
118  }
119 
120  /* Hare Move: y(i+1) = f(f(y(i))) */
121  y = __FACT_PRIMES__.mod_pow(y, 2, n) + c;
122  if (y >= n) {
123  y -= n;
124  }
125  y = __FACT_PRIMES__.mod_pow(y, 2, n) + c;
126  if (y >= n) {
127  y -= n;
128  }
129 
130  /* check gcd of |x-y| and n */
131  divisor = std::gcd(abs(x - y), n);
132 
133  /* retry if the algorithm fails to find prime factor
134  * with chosen x and c */
135  if (divisor == n) {
136  return pollard_rho(n);
137  }
138  }
139 
140  return divisor;
141 }
