116  std::vector<std::vector<T>> permutations(const std::vector<T> &vec) {
117  std::vector<std::vector<T>> result;
118  std::vector<T> temp = vec;
119  do {
120  result.push_back(temp);
121  } while (std::next_permutation(temp.begin(), temp.end()));
122  return result;
123  }

125  template <typename T>
126  std::vector<std::vector<T>> combinations(const std::vector<T> &vec, int r) {
127  std::vector<std::vector<T>> result;
128  std::vector<bool> bitmask(vec.size(), false);
129  std::fill(bitmask.begin(), bitmask.begin() + r, true);
130  do {
131  std::vector<T> combination;
132  for (int i = 0; i < vec.size(); ++i) {
133  if (bitmask[i]) {
134  combination.push_back(vec[i]);
135  }
136  }
137  result.push_back(combination);
138  } while (std::prev_permutation(bitmask.begin(), bitmask.end()));
139  return result;
140  }
