53 void gpmp::ml::BayesClf::train(const std::vector<std::vector<double>> &data,
54  const std::vector<std::string> &labels) {
55  // Count class occurrences
56  for (const auto &label : labels) {
57  class_probs[label] += 1.0;
58  }
59 
60  // Count feature occurrences for each class
61  for (size_t i = 0; i < data.size(); ++i) {
62  const std::string &label = labels[i];
63  const std::vector<double> &features = data[i];
64 
65  class_probs[label] += 1.0;
66 
67  for (size_t j = 0; j < features.size(); ++j) {
68  feature_probs[label][j] += features[j];
69  }
70  }
71 
72  // Calculate class probabilities and feature probabilities
73  double smoothing_factor = alpha * 2.0;
74  for (const auto &entry : class_probs) {
75  const std::string &label = entry.first;
76  double class_count = entry.second;
77 
78  // Calculate class probability
79  class_probs[label] =
80  (class_count + alpha) / (data.size() + smoothing_factor);
81 
82  // Calculate feature probabilities
83  for (size_t j = 0; j < feature_probs[label].size(); ++j) {
84  feature_probs[label][j] = (feature_probs[label][j] + alpha) /
85  (class_count + smoothing_factor);
86  }
87  }
88 
89  // Calculate class log priors
90  if (fit_prior) {
91  double total = std::accumulate(
92  class_probs.begin(),
93  class_probs.end(),
94  0.0,
95  [](double sum, const auto &entry) { return sum + entry.second; });
96 
97  for (auto &entry : class_probs) {
98  entry.second /= total;
99  }
100 
101  std::transform(
102  class_probs.begin(),
103  class_probs.end(),
104  class_log_prior.begin(),
105  [total](const auto &entry) { return log(entry.second); });
106  }
107 }

110 gpmp::ml::BayesClf::predict(const std::vector<double> &new_data) const {
111  double max_prob = -std::numeric_limits<double>::infinity();
112  std::string predicted_class;
113 
114  for (const auto &entry : class_probs) {
115  const std::string &label = entry.first;
116  double probability = log(entry.second);
117 
118  for (size_t j = 0; j < new_data.size(); ++j) {
119  probability += new_data[j] * log(feature_probs.at(label).at(j));
120  }
121 
122  if (probability > max_prob) {
123  max_prob = probability;
124  predicted_class = label;
125  }
126  }
127 
128  return predicted_class;
129 }

151 void gpmp::ml::BayesBernoulli::train(
152  const std::vector<std::vector<size_t>> &data,
153  const std::vector<std::string> &labels) {
154  size_t numInstances = data.size();
155  size_t num_feats = data[0].size();
156 
157  for (size_t i = 0; i < numInstances; ++i) {
158  std::string classLabel = labels[i];
159 
160  // update class probabilities
161  class_probs[classLabel] += 1.0;
162 
163  // update feature probabilities
164  for (size_t j = 0; j < num_feats; ++j) {
165  feat_probs[classLabel][j] += data[i][j];
166  }
167  }
168 
169  // laplace smoothing
170  double smoothing_factor = alpha * 2.0;
171  for (auto &entry : class_probs) {
172  entry.second =
173  (entry.second + alpha) / (numInstances + smoothing_factor);
174  }
175 
176  for (auto &class_entry : feat_probs) {
177  for (auto &feat_entry : class_entry.second) {
178  feat_entry.second =
179  (feat_entry.second + alpha) /
180  (class_probs[class_entry.first] + smoothing_factor);
181  }
182  }
183 }
