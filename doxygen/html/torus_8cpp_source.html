68void init() {
69 glClearColor(0.0, 0.0, 0.0, 1.0);
70 glColor3f(1.0, 1.0, 1.0);

72 glMatrixMode(GL_PROJECTION);

74 gluPerspective(60.0, 4.0 / 3.0, 1, 40);
