50/*
51 * handles window reshaping/resizing
52 */
53void reshape(GLint w, GLint h) {
54 glViewport(0, 0, w, h);
55 GLfloat aspect = (GLfloat)w / (GLfloat)h;
56 glMatrixMode(GL_PROJECTION);
57 glLoadIdentity();
58
59 if (w <= h) {
60 // width is smaller, go from -50 .. 50 in width
61 glOrtho(-50.0, 50.0, -50.0 / aspect, 50.0 / aspect, -1.0, 1.0);
62 }
63
64 else {
65 // height is smaller, go from -50 .. 50 in height
66 // glOrtho(-50.0*aspect, 50.0*aspect, -50.0, 50.0, -1.0, 1.0);
67
68 glOrtho(-10.0 * aspect, 10.0 * aspect, -10.0, 10.0, -1.0, 1.0);
69 }
70}

72void display() {
73 // setting every pixel in frame buffer to a clear color
74 glClear(GL_COLOR_BUFFER_BIT);
75 glMatrixMode(GL_MODELVIEW);
76 glLoadIdentity();
77
78 glColor3f(1.0, 1.0, 1.0);
79 glRotatef(current_rot, 0.0, 0.0, 1.0);
80 glutWireTorus(0.5, 3, 15, 30);
81
82 // being drawing
83
84 glBegin(GL_POLYGON);
85 glRotatef(current_rot, 1.0, 0.0, 1.0);
86 glColor3f(1, 0, 0);
87 glVertex3f(0, 0, 0);
88 glVertex3f(10, 0, 0);
89 glColor3f(0, 1, 0);
90 glVertex3f(0, 0, 0);
91 glVertex3f(0, 10, 0);
92 glColor3f(0, 0, 1);
93 glVertex3f(0, 0, 0);
94 glVertex3f(0, 0, 10);
95 glEnd();
96
97 glRotatef(current_rot, 1.0, 0.0, 1.0);
98
99 // flush the draw cmd to display image immediately
100 glFlush();
101 glutSwapBuffers();
102}

104/*
105 * timer that increments the torus orientations for animation
106 */
107void timer(int v) {
108 if (spinning) {
109 current_rot += 1.0;
110
111 if (current_rot > 360.0) {
112 current_rot -= 360.0;
113 }
114 glutPostRedisplay();
115 }
116
117 glutTimerFunc(1000 / FPS, timer, v);
118}

120/*
121 * mouse events, left click generates new animations, right clock
122 * removes idle-time callback
123 */
124void mouse(int button, int state, int x, int y) {
125 if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN) {
126 spinning = true;
127 } else if (button == GLUT_RIGHT_BUTTON && state == GLUT_DOWN) {
128 spinning = false;
129 }
130}
