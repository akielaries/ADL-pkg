44/*
45 * For this particular exmaple this is a class which contains the
46 * turnover and success of a company and the characteristics of the
47 * team. The file with the data will be a csv file whit 3 columns:
48 * - turover(int)
49 * - characteristics(unsigned)
50 * - success(boolean)
51 */
52class Company {
53 public:
54 Company(int64_t turnover, string characteristics, bool success)
55 : turnover(turnover), success(success), distance(-1) {
56 this->characteristics = transformCharacteristics(characteristics);
57 }
58
59 // distance from test point
60 double distance;
61 int64_t turnover;
62 int64_t characteristics;
63 bool success;
64
65 private:
66 /*
67 * Function that transforms characteristics (Very
68 * Strong,Strong,Average,Weak) as strings to number. It is geting
69 * the ASCII code as decimal number and retuning the total sum of
70 * all characters within the string.
71 */
72 int64_t transformCharacteristics(string &c) {
73 uint64_t sum = 0;
74 for (size_t i = 0; i < c.size(); ++i) {
75 if (c[i] != ' ') {
76 sum += (int64_t)c[i];
77 }
78 }
79 return sum;
80 }
81};

116 vector<string> split(string target, string delim) {
117 vector<string> v;
118 if (!target.empty()) {
119 size_t start = 0;
120 do {
121 size_t x = target.find(delim, start);
122 // a check whether the target is found
123 if (x == -1) {
124 break;
125 }
126 string tmp = target.substr(start, x - start);
127 v.push_back(tmp);
128 start += delim.size() + tmp.size();
129 } while (true);
130
131 v.push_back(target.substr(start));
132 }
133 return v;
134 }

142long double euclideanDistance(Company &lhs, Company &test) {
143 return sqrt(pow((lhs.turnover - test.turnover), 2) +
144 pow((lhs.characteristics - test.characteristics), 2));
145}

147long double manhattanDistance(Company &lhs, Company &test) {
148 return (abs(lhs.turnover - test.turnover) +

159bool KNN(vector<Company> &data, Company &test, int k,
160 double (*distanceFunction)(Company &, Company &)) {
161 // filling the distances between all points and test
162 fillDistances(data, test, distanceFunction);
163
164 // sorting so that we can get the k nearest
165 sort(data.begin(), data.end(), comparison);
166
167 int64_t countSuccesful = 0;
168 int64_t countUnsuccesful = 0;
169 for (int64_t i = 0; i < k; ++i) {
170 if (data[i].success) {
171 ++countSuccesful;
172 } else {
173 ++countUnsuccesful;
174 }
175 }
176 /*
177 * Based on the count of succesful and unsuccessful examples
178 * within those k, we are deciding whether the test example is
179 * successful or not.
180 */
181 if (countSuccesful == countUnsuccesful) {
182 test.success = data[0].success;
183 } else {
184 test.success = countSuccesful > countUnsuccesful ? true : false;
185 }
186 return test.success;
187}
