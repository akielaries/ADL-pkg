165 inline long double sigmoid_activ(long double x) {
166 return 1.0f / (1 + exp(-x));
167 }
168
174 inline long double sigmoid_deriv(long double x) {
175 return (x * (1 - x));
176 }

178 std::vector<size_t> layer_units;
179 std::vector<Matrix<T>> bias_vectors;
180 std::vector<Matrix<T>> wt_mtx;
181 std::vector<Matrix<T>> activations;
182
183 long double lr;
184
190 explicit SecondaryMLP(std::vector<size_t> layer_units,
191 long double lr = .001)
192 : layer_units(layer_units), wt_mtx(), bias_vectors(), lr(lr) {
193 // traverse the elements
194 for (size_t i = 0; i < layer_units.size() - 1; ++i) {
195 // size of inputs
196 size_t inputs{layer_units[i]};
197 // size of outputs
198 size_t outputs{layer_units[i + 1]};
199
200 // set to random Guassian Noise related values
201 // weights
202 auto gauss_wt = mtpk::mtx<T>::randn(outputs, inputs);
203 wt_mtx.push_back(gauss_wt);
204 // biases
205 auto bias_wt = mtpk::mtx<T>::randn(outputs, 1);
206 bias_vectors.push_back(bias_wt);
207 // activation function
208 activations.resize(layer_units.size());
209 }
210 }
217 auto prop_forwards(Matrix<T> x) {
218 assert(get<0>(x.shape) == layer_units[0] && get<1>(x.shape));
219 // input = to previously declared acitvations method
220 activations[0] = x;
221 Matrix prev(x);
222
223 // traverse layer units
224 for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
225 Matrix y = wt_mtx[i].mult(prev);
226 y = y + bias_vectors[i];
227 y = y.apply_function(sigmoid_activ);
228 activations[i + 1] = y;
229 prev = y;
230 }
231 return prev;
232 }
241 void prop_backwards(Matrix<T> target) {
242 assert(get<0>(target.shape) == layer_units.back());
243 // calculate the error, target - ouput
244 auto y = target;
245 auto y_hat = activations.back();
246 auto err = target - y_hat;
247 // back propagate the error calculated from output to input
248 // and step the weights
249 for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
250 // calculate errors for previous layer
251 auto wt = wt_mtx[i].T();
252 auto prior_errs = wt.mult(err);
253 auto outputs_d =
254 activations[i + 1].apply_function(sigmoid_deriv);
255 auto gradients = err.hadamard(outputs_d);
256 gradients = gradients.scalar_mult(lr);
257 auto trans_a = activations[i].T();
258 auto gradients_wt = gradients.mult(trans_a);
259
260 // adjust the networks weights based on propagation
261 // technique
262 bias_vectors[i] = bias_vectors[i].add(gradients);
263 wt_mtx[i] = wt_mtx[i].add(gradients_wt);
264 err = prior_errs;
265 }
266 }
267};
