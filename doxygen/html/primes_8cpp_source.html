53int64_t mtpk::Primality::mod_pow(int64_t base, int64_t exponent,
54 int64_t mod) {
55 int64_t x = 1;
56 int64_t y = base;
57
58 while (exponent > 0) {
59 if (exponent & 1) {
60 x = (x * y) % mod;
61 }
62 exponent = exponent >> 1;
63 y = (y * y) % mod;
64 }
65
66 return x;
67}

81/*
82 * determining if a given number is likely to be prime
83 */
84bool mtpk::Primality::compute_miller_rabin(int64_t d, int64_t n) {
85 // Pick a random number in [2..n-2] Corner cases make sure that n
86 // > 4
87 int64_t a = 2 + rand() % (n - 4);
88
89 // Compute a^d % n
90 int64_t x = mod_pow(a, d, n);
91
92 if (x == 1 || x == n - 1)
93 return true;
94
95 // Keep squaring x while one of the following doesn't
96 // happen
97 // (I) d does not reach n-1
98 // (II) (x^2) % n is not 1
99 // (III) (x^2) % n is not n-1
100 while (d != n - 1) {
101 x = (x * x) % n;
102 d *= 2;
103
104 if (x == 1)
105 return false;
106 if (x == n - 1)
107 return true;
108 }
109
110 // Return composite
111 return false;
112}

114bool mtpk::Primality::miller_rabin_prime(int64_t n, int64_t iters) {
115 /*
116 * this method will return true if n is prime, iters will
117 * determine accuracy
118 */
119 // these are corner cases
120 if (n <= 1 || n == 4)
121 return false;
122 if (n <= 3)
123 return true;
124
125 // odd will represent an odd number
126 int64_t odd = n - 1;
127 while (odd % 2 == 0) {
128 odd /= 2;
129 }
130
131 // iterate given the iters paramater
132 for (int64_t i = 0; i < iters; i++) {
133 // compute using the Miller-Rabin method
134 if (!compute_miller_rabin(odd, n)) {
135 return false;
136 }
137 }
138 return true;
139}

159/*
160 * another algorithm capable of finding primes
161 */
162int64_t mtpk::Primality::jacobian_number(int64_t a, int64_t n) {
163 if (!a)
164 return 0; // (0/n) = 0
165
166 int64_t ans = 1;
167
168 if (a < 0) {
169 a = -a; // (a/n) = (-a/n)*(-1/n)
170 if (n % 4 == 3)
171 ans = -ans; // (-1/n) = -1 if n = 3 (mod 4)
172 }
173
174 if (a == 1)
175 return ans; // (1/n) = 1
176
177 while (a) {
178 if (a < 0) {
179 a = -a; // (a/n) = (-a/n)*(-1/n)
180 if (n % 4 == 3)
181 ans = -ans; // (-1/n) = -1 if n = 3 (mod 4)
182 }
183
184 while (a % 2 == 0) {
185 a = a / 2;
186 if (n % 8 == 3 || n % 8 == 5)
187 ans = -ans;
188 }
189 std::swap(a, n);
190
191 if (a % 4 == 3 && n % 4 == 3)
192 ans = -ans;
193
194 a = a % n;
195
196 if (a > n / 2)
197 a = a - n;
198 }
199
200 if (n == 1)
201 return ans;
202
203 return 0;
204}
205/*
206 * primality test determining if a number is composite or probably
207 * prime.
208 *
209 * a^(p-1)/2 = (a/p) (mod p)
210 */
211bool mtpk::Primality::solovoy_strassen(int64_t p, int64_t iters) {
212 if (p < 2)
213 return false;
214
215 if (p != 2 && p % 2 == 0)
216 return false;
217
218 for (int64_t i = 0; i < iters; i++) {
219 // Generate a random number a
220 int64_t a = rand() % (p - 1) + 1;
221 int64_t jacobian = (p + jacobian_number(a, p)) % p;
222 int64_t mod = mod_pow(a, (p - 1) / 2, p);
223
224 if (!jacobian || mod != jacobian)
225 return false;
226 }
227 return true;
228}

230/*
231 * Absolute Fermat Pseudoprimes referred to as Carmichael Numbers
232 * are composite integers satisfying the congruence forumla below
233 * b^n - 1 = b (mod n)
234 */
235bool mtpk::Primality::carmichael_num(int64_t n) {
236 for (int64_t b = 2; b < n; b++) {
237 // If "b" is relatively prime to n
238 if (ba.op_gcd(b, n) == 1)
239 // And pow(b, n-1)%n is not 1, return false.
240 if (mod_pow(b, n - 1, n) != 1)
241 return false;
242 }
243 return true;
244}

246/*
247 * since this module deals with a great deal of number theory and
248 * various logical arithmetic operations, the greek algorithm sieve of
249 * Eratosthenes is able to capture all prime numbers to any given
250 * limit
251 */
252void mtpk::Primality::sieve_of_eratosthenes(int64_t n) {
253 // Create a boolean array "prime[0..n]" and initialize
254 // all entries it as true. A value in prime[i] will
255 // finally be false if i is Not a prime, else true.
256 bool prime[n + 1];
257 memset(prime, true, sizeof(prime));
258
259 for (int64_t p = 2; p * p <= n; p++) {
260 // If prime[p] is not changed, then it is a prime
261 if (prime[p] == true) {
262 // Update all multiples of p greater than or
263 // equal to the square of it numbers which are
264 // multiple of p and are less than p^2 are
265 // already been marked.
266 for (int64_t i = p * p; i <= n; i += p)
267 prime[i] = false;
268 }
269 }
270
271 // Print all prime numbers
272 for (int64_t p = 2; p <= n; p++)
273 if (prime[p])
274 std::cout << p << " " << std::endl;
275}

277/*
278 * algorithm for integer factorization proportial to the runtime of
279 * the square root of the size of the smallest prime factor of the
280 * composite factor being factorized. Given a positive and composite
281 * integer n, find a divisor of it.
282 *
283 * x_n-1 = x^2_n + a(mod n)
284 *
285 * n = 1111, set x_0 = 2 & f(x) = x^2 + 1
286 *
287 * x_1 = 5
288 * x_2 = 26 gcd(26 - 5, 1111) = 1
289 * x_3 = 677
290 * x_4 = 598 gcd(698 - 26, 1111) = 11
291 */
292int64_t mtpk::Primality::pollard_rho(int64_t n) {
293 /* initialize random seed */
294 srand(time(NULL));
295
296 /* no prime divisor for 1 */
297 if (n == 1)
298 return n;
299
300 /* even number means one of the divisors is 2 */
301 if (n % 2 == 0)
302 return 2;
303
304 /* we will pick from the range [2, N) */
305 int64_t x = (rand() % (n - 2)) + 2;
306 int64_t y = x;
307
308 /* the constant in f(x).
309 * Algorithm can be re-run with a different c
310 * if it throws failure for a composite. */
311 int64_t c = (rand() % (n - 1)) + 1;
312
313 /* Initialize candidate divisor (or result) */
314 int64_t divisor = 1;
315
316 /* until the prime factor isn't obtained.
317 If n is prime, return n */
318 while (divisor == 1) {
319 /* Tortoise Move: x(i+1) = f(x(i)) */
320 // take power
321 // calculate modulus
322 x = (mod_pow(x, 2, n) + c + n) % n;
323
324 /* Hare Move: y(i+1) = f(f(y(i))) */
325 y = (mod_pow(y, 2, n) + c + n) % n;
326 y = (mod_pow(y, 2, n) + c + n) % n;
327
328 /* check gcd of |x-y| and n */
329 divisor = ba.op_gcd(abs(x - y), n);
330 // divisor = std::__gcd(abs(x - y), n);
331 /* retry if the algorithm fails to find prime factor
332 * with chosen x and c */
333 if (divisor == n)
334 return pollard_rho(n);
335 }
336
337 return divisor;
338}

340/*
341 * Euler's Totient Function counts the positive numbers up until a
342 * given integer
343 *
344 * Definition: For any positive integer n, Î¦(n) is the number of all
345 * positive integers less than or equal to n that are relatively prime
346 * to n.
347 *
348 * Example: Î¦(1) = 1
349 * Î¦(2) = 1
350 * Î¦(3) = 2
351 * Î¦(4) = 2
352 * Î¦(5) = 4
353 */
354int64_t mtpk::Primality::ETF(uint64_t n) {
355 uint64_t result = 1;
356
357 for (int64_t index = 2; uint64_t(index) < n; index++) {
358 if (ba.op_gcd(index, n) == 1) {
359 result++;
360 }
361 }
362
363 return result;
364}
