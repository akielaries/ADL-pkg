43 void gpmp::ml::SVC::fit(const std::vector<std::vector<double>> &X_train,
44  const std::vector<int> &y_train) {
45  // Initialize weights and bias
46  weights.resize(X_train[0].size(), 0.0);
47  bias = 0.0;
48 
49  // Stochastic Gradient Descent
50  for (int iter = 0; iter < max_iterations; ++iter) {
51  update_weights(X_train, y_train);
52 
53  // Check convergence
54  double loss = compute_loss(X_train, y_train);
55  if (loss < tolerance) {
56  break;
57  }
58  }

80 double gpmp::ml::SVC::compute_loss(const std::vector<std::vector<double>> &X,
81  const std::vector<int> &y) {
82  double loss = 0.0;
83  for (size_t i = 0; i < X.size(); ++i) {
84  double prediction = 0.0;
85  for (size_t j = 0; j < X[i].size(); ++j) {
86  prediction += X[i][j] * weights[j];
87  }
88  prediction += bias;
89  loss += hinge_loss(prediction, y[i]);
90  }
91  // Add L2 regularization
92  for (double weight : weights) {
93  loss += 0.5 * C * weight * weight;
94  }
95  return loss / X.size();
96 }

98 void gpmp::ml::SVC::update_weights(const std::vector<std::vector<double>> &X,
99  const std::vector<int> &y) {
100  for (size_t i = 0; i < X.size(); ++i) {
101  double prediction = 0.0;
102  for (size_t j = 0; j < X[i].size(); ++j) {
103  prediction += X[i][j] * weights[j];
104  }
105  prediction += bias;
106  double loss_grad = -y[i] * (1 - prediction);
107  if (loss_grad > 0) {
108  // Update weights
109  for (size_t j = 0; j < X[i].size(); ++j) {
110  weights[j] -= learning_rate * (C * weights[j] - y[i] * X[i][j]);
111  }
112  // Update bias
113  bias -= learning_rate * y[i];
114  }
115  }
116 }

159 double gpmp::ml::SVC::cross_val_score(const std::vector<std::vector<double>> &X,
160  const std::vector<int> &y,
161  int cv) {
162  std::vector<int> fold_sizes = k_fold_indices(X.size(), cv);
163  double avg_score = 0.0;
164  for (int i = 0; i < cv; ++i) {
165  std::vector<std::vector<double>> X_train, X_valid;
166  std::vector<int> y_train, y_valid;
167  int start = 0;
168  for (int j = 0; j < cv; ++j) {
169  if (j != i) {
170  int end = start + fold_sizes[j];
171  for (int k = start; k < end; ++k) {
172  X_train.push_back(X[k]);
173  y_train.push_back(y[k]);
174  }
175  } else {
176  int end = start + fold_sizes[j];
177  for (int k = start; k < end; ++k) {
178  X_valid.push_back(X[k]);
179  y_valid.push_back(y[k]);
180  }
181  }
182  start += fold_sizes[j];
183  }
184  fit(X_train, y_train);
185  double score_val = score(X_valid, y_valid);
186  if (verbose) {
187  std::cout << "Cross-validation fold " << i + 1
188  << " accuracy: " << score_val << std::endl;
189  }
190  avg_score += score_val;
191  }
192  return avg_score / cv;
193 }
