264  return n;

267  if (n % 2 == 0)
268  return 2;

272  long long int y = x;

275  * Algorithm can be re-run with a different c

277  long long int c = (rand() % (n - 1)) + 1;
278 
279  /* Initialize candidate divisor (or result) */
280  long long int divisor = 1;
281 
282  /* until the prime factor isn't obtained.
283  If n is prime, return n */
284  while (divisor == 1) {
285  /* Tortoise Move: x(i+1) = f(x(i)) */
286  // take power
287  // calculate modulus
288  x = (mod_pow(x, 2, n) + c + n) % n;
289 
290  /* Hare Move: y(i+1) = f(f(y(i))) */
291  y = (mod_pow(y, 2, n) + c + n) % n;
292  y = (mod_pow(y, 2, n) + c + n) % n;
