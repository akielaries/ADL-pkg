51  A += S[0];

53  for (uint32_t i = 1; i <= r; ++i) {
54  A = rotl((A ^ B), B) + S[2 * i];

60  for (uint32_t i = r; i > 0; --i) {
61  B = rotr(B - S[2 * i + 1], A) ^ A;
62  A = rotr(A - S[2 * i], B) ^ B;
63  }
64  B -= S[1];
65  A -= S[0];
